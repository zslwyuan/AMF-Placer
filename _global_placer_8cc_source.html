610void GlobalPlacer::updatePseudoNetWeight(float &pseudoNetWeight, int curIter)
611{
612 progressRatio = lowerBoundHPWL / upperBoundHPWL;
613 if (progressRatio > 1)
614 progressRatio = 0.999;
615 if (progressRatio > 0.5)
616 spreadingForgetRatio = 1 - 0.8 * progressRatio;
617 else // the spreading control is not worthy when the placement is far from convergence.
618 spreadingForgetRatio = 1;
619 if (disableSpreadingForgetRatio)
620 spreadingForgetRatio = 1;
621 HPWLChangeLittle = false;
622 progressRatio = std::pow(progressRatio, 0.6); // 0.5-1.4;0.6-1.3
623 placementInfo->setProgress(progressRatio);
624
625 if (pseudoNetWeight < 0.0020)
626 pseudoNetWeight *= 1.825;
627 else
628 pseudoNetWeight *= 1.5 * (1 - progressRatio) + 1.01 * (progressRatio);
629
630 historyHPWLs.push_back(upperBoundHPWL);
631
632 int numRecorded = historyHPWLs.size();
633 if (numRecorded > 5)
634 {
635 float err = 0;
636 for (int vecId = numRecorded - 5; vecId < numRecorded; vecId++)
637 {
638 err += (minHPWL - historyHPWLs[vecId]) * (minHPWL - historyHPWLs[vecId]);
639 }
640 if (historyHPWLs[numRecorded - 1] > historyHPWLs[numRecorded - 2] &&
641 historyHPWLs[numRecorded - 2] > historyHPWLs[numRecorded - 3])
642 {
643 if (pseudoNetWeight > 0.03 && progressRatio < 0.95)
644 {
645 while (historyHPWLs.size() > 1)
646 historyHPWLs.pop_front();
647 pseudoNetWeight *= 0.75;
648 }
649 }
650 else if (progressRatio > 0.8 && upperBoundHPWL / minHPWL > 1.333 && updateMinHPWLAfterLegalization)
651 {
652 print_warning("The upperbound HPWL increases too much, pseudoNetWeight is reduced by 50% to recover.");
653 placementInfo->adjustLUTFFUtilization_Routability_Reset();
654 pseudoNetWeight *= 0.5;
655 historyHPWLs.clear();
656 }
657 err = std::sqrt(err / 5);
658 print_info("err=" + std::to_string(err) + " minHPWL=" + std::to_string(minHPWL) +
