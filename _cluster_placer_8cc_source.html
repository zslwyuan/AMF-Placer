132void ClusterPlacer::createLongPathClusterUnits()
133{
134 // handle the long paths
135 auto &timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo_PathLenSorted();
136 // auto simpleTimingGraph = placementInfo->getTimingInfo()->getSimplePlacementTimingGraph();
137 int pathLengthThr = placementInfo->getLongPathThresholdLevel();
138 std::set<int> extractedCellIds;
139 extractedCellIds.clear();
140
141 print_status("ClusterPlacer:: create long path cluster units (pathLengthThr=" + std::to_string(pathLengthThr) +
142 ")");
143 unsigned int maxSize = 0;
144 for (auto timingNode : timingNodes)
145 {
146 if (timingNode->getLongestPathLength() > pathLengthThr && timingNode->getForwardLevel() > pathLengthThr * 0.333)
147 {
148 auto curCell = timingNode->getDesignNode();
149 auto tmpPU = placementInfo->getPlacementUnitByCell(curCell);
150 if (placementUnitId2ClusterUnitId[tmpPU->getId()] < 0)
151 {
152 std::vector<int> candidateCellIds; //= simpleTimingGraph->BFSFromNode(timingNode->getId(),
153 // pathLengthThr, 1000000, extractedCellIds);
154 candidateCellIds.clear();
155
156 if (timingNode->getOutEdges().size() >= 16)
157 {
158
159 for (auto outEdge : timingNode->getOutEdges())
160 {
161 if (extractedCellIds.find(outEdge->getSink()->getId()) == extractedCellIds.end())
162 candidateCellIds.push_back(outEdge->getSink()->getId());
163 }
164
165 // if (candidateCellIds.size() < 16)
166 // continue;
167 }
168 else
169 {
170 continue;
171 }
172
173 std::set<PlacementInfo::PlacementUnit *> PUsInLongPaths;
174 PUsInLongPaths.clear();
175 for (auto &candidateCellId : candidateCellIds)
176 {
177 extractedCellIds.insert(candidateCellId);
178 auto PUInPath = placementInfo->getPlacementUnitByCellId(candidateCellId);
179 // the clusters should not be overlapped with each other
180 if (placementUnitId2ClusterUnitId[PUInPath->getId()] < 0)
181 PUsInLongPaths.insert(PUInPath);
182 }
183
184 if (PUsInLongPaths.size() == 0)
185 continue;
186

188 for (auto PUInPath : PUsInLongPaths)

190 assert(placementUnitId2ClusterUnitId[PUInPath->getId()] < 0);
191 placementUnitId2ClusterUnitId[PUInPath->getId()] = curClusterUnit->getId();
192 curClusterUnit->addPlacementUnit(PUInPath);
193
194 if (!PUInPath->isFixed())
195 {
196 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(PUInPath))
197 {
198 int cellId = unpackedCell->getCell()->getCellId();

200 }

207 }
208 }
209 }
210 }
211
212 if (PUsInLongPaths.size() > maxSize)
213 maxSize = PUsInLongPaths.size();
214
215 clusterUnits.push_back(curClusterUnit);
216 }
217 }
218 assert(clusterUnits.size() <= placementInfo->getPlacementUnits().size());
219 }
220 print_info("ClusterPlacer: largest long-path cluster size=" + std::to_string(maxSize));

445void ClusterPlacer::refineClustersWithPredefinedClusters()
446{
447 std::vector<std::vector<DesignInfo::DesignCell *>> &predefinedCellClusters =
448 placementInfo->getDesignInfo()->getPredefinedClusters();
449
450 std::set<PlacementInfo::PlacementUnit *> reallocatedPUs;
451
452 int reallocateCnt = 0;
453 for (auto &clusterCellSet : predefinedCellClusters)
454 {
455 std::vector<int> cluster2overlappedCellNum(clusters.size(), 0);
456 int maxOverlappedClusterId = 0;
457 int maxOverlapCnt = 0;
458
459 int DSPCnt = 0;
460 int BRAMCnt = 0;
461 for (auto tmpCell : clusterCellSet)
462 {
463 DSPCnt += tmpCell->isDSP();
464 BRAMCnt += tmpCell->isBRAM();
465 }
466
467 if (DSPCnt >= 8 || BRAMCnt >= 8)
468 continue;
469
470 // find the most overlapped partitioned cluster for the predefined cluster
471 for (auto tmpCell : clusterCellSet)
472 {
473 auto tmpPU = placementInfo->getPlacementUnitByCell(tmpCell);
474 auto tmpPUId = tmpPU->getId();
475
476 for (unsigned int i = 0; i < clusters.size(); i++)
477 {
478 if (clusters[i].find(tmpPUId) != clusters[i].end())
479 {
480 cluster2overlappedCellNum[i]++;
481 if (cluster2overlappedCellNum[i] > maxOverlapCnt)
482 {
483 cluster2overlappedCellNum[i] = cluster2overlappedCellNum[i];
484 maxOverlappedClusterId = i;
485 }
486 break;
487 }
488 }
489 }
490
491 // move cells in predefined Cluster togather
492 for (auto tmpCell : clusterCellSet)
493 {
494 auto tmpPU = placementInfo->getPlacementUnitByCell(tmpCell);
495 auto tmpPUId = tmpPU->getId();
496
497 // don't reallocate PU twice
498 if (reallocatedPUs.find(tmpPU) != reallocatedPUs.end())
499 continue;
500 reallocatedPUs.insert(tmpPU);
501
502 if (placementUnit2ClusterId[tmpPUId] != maxOverlappedClusterId)
503 {
504 clusters[placementUnit2ClusterId[tmpPUId]].erase(tmpPUId);
505 clusters[maxOverlappedClusterId].insert(tmpPUId);
506 placementUnit2ClusterId[tmpPUId] = maxOverlappedClusterId;
507 reallocateCnt++;
508 }
509 }
510 }
511
512 print_info("reallocate " + std::to_string(reallocateCnt) + " PU(s)");
513}

581void ClusterPlacer::setClusterNetsAdjMat()
582{
583 clusterAdjMat = std::vector<std::vector<float>>(clusters.size(), std::vector<float>(clusters.size(), 0));
584 clusterCLBCellWeights = std::vector<float>(clusters.size(), 0);
585 clusterDSPCellWeights = std::vector<float>(clusters.size(), 0);
586 clusterBRAMCellWeights = std::vector<float>(clusters.size(), 0);
587 cluster2FixedUnitMat = std::vector<std::vector<float>>(
588 clusters.size(), std::vector<float>(placementInfo->getFixedPlacementUnits().size(), 0));
589
590 fixedX.clear();
591 fixedY.clear();
592
593 float clockRegionW = (placementInfo->getGlobalMaxX() - placementInfo->getGlobalMinX()) / clockRegionXNum;
594
595 std::map<PlacementInfo::PlacementUnit *, int> fixedPlacementUnit2LocId;
596 fixedPlacementUnit2LocId.clear();
597
598 float pathThr = placementInfo->getMediumPathThresholdLevel();
599
600 int ignoredConnectCnt = 0;
601 bool cutShortPathNets = !isDensePlacement();
602 for (auto net : placementInfo->getPlacementNets())
603 {
604 for (auto driveU : net->getDriverUnits())
605 {
606 if (driveU->isFixed())
607 {
608 if (fixedPlacementUnit2LocId.find(driveU) == fixedPlacementUnit2LocId.end())
609 {
610 int amo = fixedPlacementUnit2LocId.size();
611 fixedPlacementUnit2LocId[driveU] = amo;
612 if (containIOCells(driveU))
613 {
614 fixedX.push_back(driveU->X() + clockRegionW / 2.0);
615 fixedY.push_back(driveU->Y());
616 }
617 else
618 {
619 fixedX.push_back(driveU->X());
620 fixedY.push_back(driveU->Y());
621 }
622 }
623 }
624
625 for (auto UBeDriven : net->getUnitsBeDriven())
626 {
627 int A = driveU->getId();
628 int B = UBeDriven->getId();
629 int maxLength = getPlacementUnitMaxPathLen(UBeDriven);
630
631 if (maxLength > 0 && maxLength < 3 && net->getUnitsBeDriven().size() < 8 && cutShortPathNets)
632 continue;
633
634 int clusterA = placementUnit2ClusterId[A];
635 int clusterB = placementUnit2ClusterId[B];
636 assert(clusterA >= 0 && clusterB >= 0);
637 assert((unsigned int)clusterA < cluster2FixedUnitMat.size() &&
638 (unsigned int)clusterB < cluster2FixedUnitMat.size());
639 if (UBeDriven->isFixed())
640 {
641 if (fixedPlacementUnit2LocId.find(UBeDriven) == fixedPlacementUnit2LocId.end())
642 {
643 int amo = fixedPlacementUnit2LocId.size();
644 fixedPlacementUnit2LocId[UBeDriven] = amo;
645 if (containIOCells(UBeDriven))
646 {
647 fixedX.push_back(UBeDriven->X() + clockRegionW / 2.0);
648 fixedY.push_back(UBeDriven->Y());
649 }
650 else
651 {
652 fixedX.push_back(UBeDriven->X());
653 fixedY.push_back(UBeDriven->Y());
654 }
655 }
656 }
657
658 if (net->getPinOffsetsInUnit().size() > 1)
659 {
660 if (UBeDriven->isFixed() && !driveU->isFixed())
661 {
662 assert(fixedPlacementUnit2LocId.find(UBeDriven) != fixedPlacementUnit2LocId.end());
663 assert((unsigned int)fixedPlacementUnit2LocId[UBeDriven] <
664 cluster2FixedUnitMat[clusterA].size());
665 cluster2FixedUnitMat[clusterA][fixedPlacementUnit2LocId[UBeDriven]] +=
666 net->getDesignNet()->getOverallEnhanceRatio() / (net->getPinOffsetsInUnit().size() - 1);
667 }
668 else if (!UBeDriven->isFixed() && driveU->isFixed())
669 {
670 assert(fixedPlacementUnit2LocId.find(driveU) != fixedPlacementUnit2LocId.end());
671 assert((unsigned int)fixedPlacementUnit2LocId[driveU] < cluster2FixedUnitMat[clusterB].size());
672 cluster2FixedUnitMat[clusterB][fixedPlacementUnit2LocId[driveU]] +=
673 net->getDesignNet()->getOverallEnhanceRatio() / (net->getPinOffsetsInUnit().size() - 1);
674 }
675 else if (!UBeDriven->isFixed() && !driveU->isFixed())
676 {
677 clusterAdjMat[clusterA][clusterB] +=
678 net->getDesignNet()->getOverallEnhanceRatio() / (net->getPinOffsetsInUnit().size() - 1);
679 clusterAdjMat[clusterB][clusterA] +=
680 net->getDesignNet()->getOverallEnhanceRatio() / (net->getPinOffsetsInUnit().size() - 1);
681 }
682 }
683 }
684 }
685 }
686
687 for (auto curPU : placementInfo->getPlacementUnits())
688 {
