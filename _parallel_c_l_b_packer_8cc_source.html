881std::vector<DeviceInfo::DeviceSite *> *
882ParallelCLBPacker::findNeiborSitesFromBinGrid(DesignInfo::DesignCellType curCellType, float targetX, float targetY,
883 float displacementLowerbound, float displacementUpperbound,
884 float y2xRatio)
885{
886 assert(displacementLowerbound < displacementUpperbound);
887 // please note that the input DesignCell is only used to find the corresponding binGrid for site search.
888 std::vector<DeviceInfo::DeviceSite *> *res = new std::vector<DeviceInfo::DeviceSite *>();
889 res->clear();
890
891 int binIdX, binIdY;
892 placementInfo->getGridXY(targetX, targetY, binIdX, binIdY);
893
894 auto sharedTypeIds = placementInfo->getPotentialBELTypeIDs(curCellType);
895
896 int clockRegionX, clockRegionY;
897 placementInfo->getDeviceInfo()->getClockRegionByLocation(targetX, targetY, clockRegionX, clockRegionY);
898
899 for (auto sharedTypeId : sharedTypeIds)
900 {
901 std::vector<std::vector<PlacementInfo::PlacementBinInfo *>> &curBinGrid =
902 placementInfo->getBinGrid(sharedTypeId);
903 assert(binIdY >= 0);
904 assert((unsigned int)binIdY < curBinGrid.size());
905 assert(binIdX >= 0);
906 assert((unsigned int)binIdX < curBinGrid[binIdY].size());
907
908 std::queue<std::pair<int, int>> binXYqueue;
909 std::set<std::pair<int, int>> reachedBinXYs;
910 binXYqueue.emplace(binIdX, binIdY);
911 reachedBinXYs.emplace(binIdX, binIdY);
912
913 while (binXYqueue.size() > 0)
914 {
915 std::pair<int, int> curXY = binXYqueue.front();
916 binXYqueue.pop();
917 int curbinIdX = curXY.first, curbinIdY = curXY.second;
918
919 PlacementInfo::PlacementBinInfo *curBin = curBinGrid[curbinIdY][curbinIdX];
920 float bin2TargetXYDistance = curBin->getManhattanDistanceTo(targetX, targetY);
921 if (bin2TargetXYDistance > displacementUpperbound)
922 continue;
923 for (auto tmpSite : curBin->getCorrespondingSites())
924 {
925 float tmpPUDis = fabs(targetX - tmpSite->X()) + y2xRatio * fabs(targetY - tmpSite->Y());
926 if (tmpPUDis > displacementLowerbound && tmpPUDis <= displacementUpperbound)
927 {
928 int SiteClockRegionX, SiteClockRegionY;
929 placementInfo->getDeviceInfo()->getClockRegionByLocation(tmpSite->X(), targetY, SiteClockRegionX,
930 SiteClockRegionY);
931 if (SiteClockRegionX == clockRegionX)
932 res->push_back(tmpSite);
933 }
934 }
935
936 for (int nextY = curbinIdY - 1; nextY <= curbinIdY + 1; nextY++)
937 {
938 for (int nextX = curbinIdX - 1; nextX <= curbinIdX + 1; nextX++)
939 {
940 if (!(nextY >= 0))
941 continue;
942 if (!((unsigned int)nextY < curBinGrid.size()))
943 continue;
944 if (!(nextX >= 0))
945 continue;
946 if (!((unsigned int)nextX < curBinGrid[binIdY].size()))
947 continue;
948 PlacementInfo::PlacementBinInfo *nextBin = curBinGrid[nextY][nextX];
949 float nextBin2TargetXYDistance = nextBin->getManhattanDistanceTo(targetX, targetY);
950 if (nextBin2TargetXYDistance > displacementUpperbound)
951 continue;
952 std::pair<int, int> nextXY(nextX, nextY);
953 if (reachedBinXYs.find(nextXY) == reachedBinXYs.end())
954 {
955 reachedBinXYs.insert(nextXY);
956 binXYqueue.push(nextXY);
957 }
958 }
959 }
960 }
961 }
962
963 return res;
964}
