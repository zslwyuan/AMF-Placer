329 inline int getPlacementUnitMaxPathLen(PlacementInfo::PlacementUnit *curPU)
330 {
331
332 auto &timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo();
333 if (auto unpacked = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(curPU))
334 {
335 if (unpacked->getCell()->isVirtualCell())
336 return 0;
337 return timingNodes[unpacked->getCell()->getCellId()]->getLongestPathLength();
338 }
339 else if (auto tmpMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(curPU))
340 {
341 int maxLen = 0;
342 for (auto tmpCell : tmpMacro->getCells())
343 {
344 if (tmpCell->isVirtualCell())
345 continue;
346 int len = timingNodes[tmpCell->getCellId()]->getLongestPathLength();
347 if (len > maxLen)
348 maxLen = len;
349 }
350 return maxLen;
351 }
352 return 0;
353 }
