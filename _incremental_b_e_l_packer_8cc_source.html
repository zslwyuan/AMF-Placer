255 {

258 FFpoints.clear();
259
260 for (auto curFF : FFs)
261 {
262 auto tmpPU = cellId2PlacementUnit[curFF->getCellId()];
263 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(tmpPU))
264 {
265 FFpoints.emplace_back(unpackedCell);
266 FFCouldBePackedCnt++;

268 }
269
270 // build k-d tree
271 kdt::KDTree<FFLocation> kdtree(FFpoints, y2xRatio);
272
273 for (auto &FFTmpPoint : FFpoints)
274 {
275 auto FF0 = FFTmpPoint.getUnpackedCell();
276
277 if (packedCells.find(FF0) != packedCells.end())
278 continue;
279 // K-nearest neighbor search (gets indices to neighbors)
280 int k = 10;
281 FFLocation query(FF0);
282 std::vector<int> indices = kdtree.knnSearch(query, k);
283 int closestInd = -1;
284 float closetDis = 10000000;
285 for (auto tmpInd : indices)
286 {
287 auto tmpPU1 = FFpoints[tmpInd].getUnpackedCell();
288 if (auto FF1 = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(tmpPU1))
289 {
290 if (packedCells.find(FF1) != packedCells.end())
291 continue;
292 if (FF0 == FF1)
293 continue;

295 DesignInfo::DesignCell *FFCell1 = FF1->getCell();
296 float dis = getCellDistance(cellLoc[FFCell0->getCellId()], cellLoc[FFCell1->getCellId()]);
297 if (dis < disThreshold)
298 {
299 if (dis < closetDis)
300 {
301 closetDis = dis;
302 closestInd = tmpInd;
303 }
304 }
305 }
306 }
307
308 if (closestInd >= 0)
309 {
310 auto FF1 = FFpoints[closestInd].getUnpackedCell();
311 DesignInfo::DesignCell *FFCell0 = FF0->getCell();
312 DesignInfo::DesignCell *FFCell1 = FF1->getCell();
313 FF_FFPairs.emplace_back(FFCell0, FFCell1);
314 FF0->setPacked();

324 cellLoc[FFCell1->getCellId()].X * FFCell1->getPins().size()) /

326 float newY = (cellLoc[FFCell0->getCellId()].Y * FFCell0->getPins().size() +
