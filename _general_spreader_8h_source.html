361 inline bool smartFindExpandDirection(std::set<PlacementInfo::PlacementBinInfo *> &coveredBinSet)
362 {
363 int possibleCnt = 0;
364
365 // iterate the 4 direction
366 for (int tmpDir = 0; tmpDir < 4; tmpDir++)
367 {
368 // initialize the expansion result (boundary, utlization, capacity, legal or not)
369 unsigned int newTopBinY = topBinY + expandOps[tmpDir].topYOp;
370 int newBottomBinY = bottomBinY + expandOps[tmpDir].bottomYOp;
371 int newLeftBinX = leftBinX + expandOps[tmpDir].leftXOp;
372 unsigned int newRightBinX = rightBinX + expandOps[tmpDir].rightXOp;
373 dir2utilization[tmpDir] = 1e-1;
374 dir2capacity[tmpDir] = 1e-5;
375 legalDir[tmpDir] = false;
376 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
377 {
378 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
379 {
380 // don't include the bins in other existing SpreadRegion
381 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
382 static_cast<dirType>(tmpDir), coveredBinSet))
383 {
384 // evaluate the utilization and capacity
385 getDirCapacityAndUtilization(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
386 static_cast<dirType>(tmpDir), dir2utilization[tmpDir],
387 dir2capacity[tmpDir]);
388
389 // capacity greater than 0 => legal direction
390 if (dir2capacity[tmpDir] > 2e-5)
391 {
392 actualExpandOps[tmpDir] = expandOps[tmpDir];
393 legalDir[tmpDir] = true;
394 possibleCnt++;
395 }
396
397 // if the capacity is legal, then try to further expand 1 more bin step in this direction
398 if (!legalDir[tmpDir])
399 {
400 actualExpandOps[tmpDir] = expandOps[tmpDir] + 1; // <== further expand
401 unsigned int newTopBinY = topBinY + actualExpandOps[tmpDir].topYOp;
402 int newBottomBinY = bottomBinY + actualExpandOps[tmpDir].bottomYOp;
403 int newLeftBinX = leftBinX + actualExpandOps[tmpDir].leftXOp;
404 unsigned int newRightBinX = rightBinX + actualExpandOps[tmpDir].rightXOp;
405 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
406 {
407 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
408 {
409 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
410 static_cast<dirType>(tmpDir), coveredBinSet))
411 {
412 getDirCapacityAndUtilization(newTopBinY, newBottomBinY, newLeftBinX,
413 newRightBinX, static_cast<dirType>(tmpDir),
414 dir2utilization[tmpDir], dir2capacity[tmpDir]);
415 if (dir2capacity[tmpDir] > 2e-5)
416 {
417 legalDir[tmpDir] = true;
418 possibleCnt++;
419 }
420
421 // if the capacity is legal, then try to further expand 1 more bin step in
422 // this direction
423 if (!legalDir[tmpDir])
424 {
425 actualExpandOps[tmpDir] =
426 (expandOps[tmpDir] + 1) + 1; // <== further expand
427 unsigned int newTopBinY = topBinY + actualExpandOps[tmpDir].topYOp;
428 int newBottomBinY = bottomBinY + actualExpandOps[tmpDir].bottomYOp;
429 int newLeftBinX = leftBinX + actualExpandOps[tmpDir].leftXOp;
430 unsigned int newRightBinX =
431 rightBinX + actualExpandOps[tmpDir].rightXOp;
432 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
433 {
434 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
435 {
436 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX,
437 newRightBinX, static_cast<dirType>(tmpDir),
438 coveredBinSet))
439 {
440 getDirCapacityAndUtilization(
441 newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
442 static_cast<dirType>(tmpDir), dir2utilization[tmpDir],
443 dir2capacity[tmpDir]);
444 if (dir2capacity[tmpDir] > 2e-5)
445 {
446 legalDir[tmpDir] = true;
447 possibleCnt++;
448 }
449 }
450 }
451 }
452 }
453 }
454 }
455 }
456 }
457 }
458 }
459 }
460 }
461 return possibleCnt;

897 {

912 }

948 {
949 // Random selection of pivot.
950 int pvt, n;
951 n = HiFPlacer_hashprimes[low & 0xff];
952 pvt = low + n % (high - low + 1); // Randomizing the pivot value from sub-array.
953 std::swap(cellIds[high], cellIds[pvt]);
