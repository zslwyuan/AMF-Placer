363 inline bool smartFindExpandDirection(std::set<PlacementInfo::PlacementBinInfo *> &coveredBinSet)
364 {
365 int possibleCnt = 0;
366
367 // iterate the 4 direction
368 for (int tmpDir = 0; tmpDir < 4; tmpDir++)
369 {
370 // initialize the expansion result (boundary, utlization, capacity, legal or not)
371 unsigned int newTopBinY = topBinY + expandOps[tmpDir].topYOp;
372 int newBottomBinY = bottomBinY + expandOps[tmpDir].bottomYOp;
373 int newLeftBinX = leftBinX + expandOps[tmpDir].leftXOp;
374 unsigned int newRightBinX = rightBinX + expandOps[tmpDir].rightXOp;
375 dir2utilization[tmpDir] = 1e-1;
376 dir2capacity[tmpDir] = 1e-5;
377 legalDir[tmpDir] = false;
378 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
379 {
380 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
381 {
382 // don't include the bins in other existing SpreadRegion
383 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
384 static_cast<dirType>(tmpDir), coveredBinSet))
385 {
386 // evaluate the utilization and capacity
387 getDirCapacityAndUtilization(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
388 static_cast<dirType>(tmpDir), dir2utilization[tmpDir],
389 dir2capacity[tmpDir]);
390
391 // capacity greater than 0 => legal direction
392 if (dir2capacity[tmpDir] > 2e-5)
393 {
394 actualExpandOps[tmpDir] = expandOps[tmpDir];
395 legalDir[tmpDir] = true;
396 possibleCnt++;
397 }
398
399 // if the capacity is legal, then try to further expand 1 more bin step in this direction
400 if (!legalDir[tmpDir])
401 {
402 actualExpandOps[tmpDir] = expandOps[tmpDir] + 1; // <== further expand
403 unsigned int newTopBinY = topBinY + actualExpandOps[tmpDir].topYOp;
404 int newBottomBinY = bottomBinY + actualExpandOps[tmpDir].bottomYOp;
405 int newLeftBinX = leftBinX + actualExpandOps[tmpDir].leftXOp;
406 unsigned int newRightBinX = rightBinX + actualExpandOps[tmpDir].rightXOp;
407 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
408 {
409 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
410 {
411 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
412 static_cast<dirType>(tmpDir), coveredBinSet))
413 {
414 getDirCapacityAndUtilization(newTopBinY, newBottomBinY, newLeftBinX,
415 newRightBinX, static_cast<dirType>(tmpDir),
416 dir2utilization[tmpDir], dir2capacity[tmpDir]);
417 if (dir2capacity[tmpDir] > 2e-5)
418 {
419 legalDir[tmpDir] = true;
420 possibleCnt++;
421 }
422
423 // if the capacity is legal, then try to further expand 1 more bin step in
424 // this direction
425 if (!legalDir[tmpDir])
426 {
427 actualExpandOps[tmpDir] =
428 (expandOps[tmpDir] + 1) + 1; // <== further expand
429 unsigned int newTopBinY = topBinY + actualExpandOps[tmpDir].topYOp;
430 int newBottomBinY = bottomBinY + actualExpandOps[tmpDir].bottomYOp;
431 int newLeftBinX = leftBinX + actualExpandOps[tmpDir].leftXOp;
432 unsigned int newRightBinX =
433 rightBinX + actualExpandOps[tmpDir].rightXOp;
434 if (newTopBinY < binGrid.size() && newBottomBinY >= 0)
435 {
436 if (newRightBinX < binGrid[newTopBinY].size() && newLeftBinX >= 0)
437 {
438 if (!isCovered(newTopBinY, newBottomBinY, newLeftBinX,
439 newRightBinX, static_cast<dirType>(tmpDir),
440 coveredBinSet))
441 {
442 getDirCapacityAndUtilization(
443 newTopBinY, newBottomBinY, newLeftBinX, newRightBinX,
444 static_cast<dirType>(tmpDir), dir2utilization[tmpDir],
445 dir2capacity[tmpDir]);
446 if (dir2capacity[tmpDir] > 2e-5)
447 {
448 legalDir[tmpDir] = true;
449 possibleCnt++;
450 }
451 }
452 }
453 }
454 }
455 }
456 }
457 }
458 }
459 }
460 }
461 }
462 }

950 {
951 // Random selection of pivot.
952 int pvt, n;
953 n = HiFPlacer_hashprimes[low & 0xff];
954 pvt = low + n % (high - low + 1); // Randomizing the pivot value from sub-array.
