52template <class NodeList, class NetList>
53void GraphPartitioner<NodeList, NetList>::recursiveMinCutPartition(
54 std::vector<int> &inputCluster, GraphPartitioner<NodeList, NetList> *graphPartitioner, int eachClusterDSPNum,
55 int eachClusterBRAMNum)
56{
57
58 std::array<std::vector<int>, 2> outputClusters;
59
60 sem_wait(&graphPartitioner->partitionSem);
61 int cut = graphPartitioner->minCutBipartition(inputCluster, outputClusters, graphPartitioner, eachClusterDSPNum,
62 eachClusterBRAMNum);
63 sem_post(&graphPartitioner->partitionSem);
64
65 if (cut == 0)
66 {
67 graphPartitioner->clustersLock.lock();
68 graphPartitioner->clusters.push_back(inputCluster);
69 graphPartitioner->clustersLock.unlock();
70 return;
71 }
72 if (outputClusters[0].size() == 0 && outputClusters[1].size() == 0)
73 {
74 graphPartitioner->clustersLock.lock();
75 graphPartitioner->clusters.push_back(inputCluster);
76 graphPartitioner->clustersLock.unlock();
77 return;
78 }
79 assert(outputClusters[0].size() > 0 && outputClusters[1].size() > 0);
80
81 std::thread t1(recursiveMinCutPartition, std::ref(outputClusters[0]), std::ref(graphPartitioner),
82 std::ref(eachClusterDSPNum), std::ref(eachClusterBRAMNum));
83 std::thread t2(recursiveMinCutPartition, std::ref(outputClusters[1]), std::ref(graphPartitioner),
84 std::ref(eachClusterDSPNum), std::ref(eachClusterBRAMNum));
85 t1.join();
86 t2.join();
87}

134 }
135 else

138 if (graphPartitioner->verbose)

174 }

181 targetPath = getExePath() + "/partitionHyperGraph";
182 if (!checkFileExist(targetPath))
183 targetPath = getExePath() + "/partitionHyperGraph";
184

188 assert(false && "target partitioning process executable should be found!");

192 new ExternalProcessFunc(targetPath, shareMemorySize * 4, graphPartitioner->verbose);
193 int *shareMemory = (int *)externalProc->getSharedMemory();
194 shareMemory[0] = numHyperNodes;
195 shareMemory[1] = numPlacementNets;
196 shareMemory[2] = numPlacementPins;
197 double final_imbal = std::max(0.5 - double(minCellClusterSize) / totalWeight, 0.4);
198 *(double *)(shareMemory + 3) = final_imbal;
199 int *cut = shareMemory + 5;
200 int *cwghts = shareMemory + 6;
201 int *xpins = cwghts + numHyperNodes;
202 int *pins = xpins + numPlacementNets + 1;
203 int *partvec = pins + numPlacementPins;
204 int *partweights = partvec + numHyperNodes;

206 int *hyperNode2placementUnit = new int[numHyperNodes];
207 for (auto instId : inputCluster)
208 hyperNode2placementUnit[placementUnit2hyperNode[instId]] = instId;
209
210 for (int i = 0; i < numHyperNodes; ++i)
211 {
212 assert((unsigned int)hyperNode2placementUnit[i] < graphPartitioner->nodeList.size());
213 cwghts[i] = graphPartitioner->nodeList[hyperNode2placementUnit[i]]->getWeight();
214 partvec[i] = -1;
215 }
216
217 // map placement nets to hypernet for PaToH
218
219 int maxCut = maxCutRate * numPlacementNets;
220
221 // hyper net to hyper nodes
222 xpins[0] = 0;
223 int indexNet = 0, indexPin = 0;
224 for (auto net : graphPartitioner->netList)
225 {
226 if (isCsdNet[net->getId()])
227 {
228 for (auto tmpPU : net->getUnits())
229 if (placementUnit2hyperNode[tmpPU->getId()] != -1)
230 pins[indexPin++] = placementUnit2hyperNode[tmpPU->getId()];
231 xpins[++indexNet] = indexPin;
232 }
233 }

239 // Postprocessing

241 outputClusters[partvec[i]].push_back(inputCluster[i]);
242
243 int totalWeight0 = 0;
244 for (auto PUId : outputClusters[0])
245 totalWeight0 += graphPartitioner->nodeList[PUId]->getWeight();
246 int totalWeight1 = 0;
247 for (auto PUId : outputClusters[1])
248 totalWeight1 += graphPartitioner->nodeList[PUId]->getWeight();
249
250 unsigned minClusterTotalWeight = std::min(totalWeight0, totalWeight1);
251 double imbal = 0.5 - double(minClusterTotalWeight) / totalWeight;
252
253 graphPartitioner->cntLock.lock();
254 if ((*cut > maxCut || ((float)outputClusters[0].size() / (float)outputClusters[1].size() < 0.666) ||
255 ((float)outputClusters[1].size() / (float)outputClusters[0].size() < 0.666)) &&

257 totalBRAMNum <= 3 * eachClusterBRAMNum))
258 {
259 // if (graphPartitioner->verbose)

261 " #net: " + std::to_string(numPlacementNets) + " max_cut: " + std::to_string(maxCut) +
262 " max_imbal: " + std::to_string(final_imbal) + " get too large cut: " + std::to_string(*cut) +
263 " failed to find cut meeting requirements of min-cut or size balance. cluster[0].size=" +
264 std::to_string(outputClusters[0].size()) +
265 " cluster[1].size=" + std::to_string(outputClusters[1].size()));
266 outputClusters[0].clear();
267 outputClusters[1].clear();
268 }
269 else
270 {
271 if (graphPartitioner->verbose)
272 print_info("partitioned iter#" + std::to_string(cnt) + " #node: " + std::to_string(numHyperNodes) +
273 " #net: " + std::to_string(numPlacementNets) + " max_cut: " + std::to_string(maxCut) +
274 " max_imbal: " + std::to_string(final_imbal) + " succeed with " +

277 "cluster[0].size=" + std::to_string(outputClusters[0].size()) +
278 " cluster[1].size=" + std::to_string(outputClusters[1].size()));
279 }
280
281 ++cnt;
282 graphPartitioner->cntLock.unlock();
283
284 int res = *cut;
285 delete[] hyperNode2placementUnit;
286 delete externalProc;
287
288 return res;
289}
