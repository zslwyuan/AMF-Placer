48template <class NodeList, class NetList>
49void GraphPartitioner<NodeList, NetList>::recursiveMinCutPartition(
50 std::vector<int> &inputCluster, GraphPartitioner<NodeList, NetList> *graphPartitioner, int eachClusterDSPNum,
51 int eachClusterBRAMNum)
52{
53
54 std::array<std::vector<int>, 2> outputClusters;
55
56 sem_wait(&graphPartitioner->partitionSem);
57 int cut = graphPartitioner->minCutBipartition(inputCluster, outputClusters, graphPartitioner, eachClusterDSPNum,
58 eachClusterBRAMNum);
59 sem_post(&graphPartitioner->partitionSem);
60
61 if (cut == 0)
62 {
63 graphPartitioner->clustersLock.lock();
64 graphPartitioner->clusters.push_back(inputCluster);
65 graphPartitioner->clustersLock.unlock();
66 return;
67 }
68 if (outputClusters[0].size() == 0 && outputClusters[1].size() == 0)
69 {
70 graphPartitioner->clustersLock.lock();
71 graphPartitioner->clusters.push_back(inputCluster);
72 graphPartitioner->clustersLock.unlock();
73 return;
74 }
75 assert(outputClusters[0].size() > 0 && outputClusters[1].size() > 0);
76
77 std::thread t1(recursiveMinCutPartition, std::ref(outputClusters[0]), std::ref(graphPartitioner),
78 std::ref(eachClusterDSPNum), std::ref(eachClusterBRAMNum));
79 std::thread t2(recursiveMinCutPartition, std::ref(outputClusters[1]), std::ref(graphPartitioner),
80 std::ref(eachClusterDSPNum), std::ref(eachClusterBRAMNum));
81 t1.join();
82 t2.join();
83}

85template <class NodeList, class NetList>
86unsigned GraphPartitioner<NodeList, NetList>::minCutBipartition(const std::vector<int> &inputCluster,
87 std::array<std::vector<int>, 2> &outputClusters,
88 GraphPartitioner<NodeList, NetList> *graphPartitioner,
89 int eachClusterDSPNum, int eachClusterBRAMNum)
90{
91
92 static int cnt = 0;
93
94 outputClusters[0].clear();
95 outputClusters[1].clear();
96 unsigned int totalWeight = 0;
97
98 int totalDSPNum = 0;
99 int totalBRAMNum = 0;
100 for (auto PUId : inputCluster)
101 {
102 totalWeight += graphPartitioner->nodeList[PUId]->getWeight();
103 totalBRAMNum += graphPartitioner->nodeList[PUId]->getBRAMNum();
104 totalDSPNum += graphPartitioner->nodeList[PUId]->getDSPNum();
105 }
106
107 // Constraints
108 unsigned minCellClusterSize = graphPartitioner->minClusterCellNum;
109
110 if (inputCluster.size() < minCellClusterSize / 5)
111 return 0;
112
113 if (totalDSPNum <= eachClusterDSPNum && totalBRAMNum <= eachClusterBRAMNum)
114 {
115 if (double(minCellClusterSize) > 0.8 * totalWeight)
116 return 0;
117 else
118 {
119 if (graphPartitioner->verbose)
120 print_info("partitioning iter#" + std::to_string(cnt) + " #curNode: " +
121 std::to_string(inputCluster.size()) + " totalWeight=" + std::to_string(totalWeight) +
122 " minCellClusterSize > 0.8 * totalWeight => SHOULD CONTINUE PARTITIONING");
123 }
124 }
125 else
126 {
127 graphPartitioner->cntLock.lock();
128 if (graphPartitioner->verbose)
129 print_info("partitioning iter#" + std::to_string(cnt) + " #curNode: " +
130 std::to_string(inputCluster.size()) + " #totalDSPNum: " + std::to_string(totalDSPNum) +
131 " #totalBRAMNum: " + std::to_string(totalBRAMNum) + " SHOULD CONTINUE PARTITIONING");
132 graphPartitioner->cntLock.unlock();
133 }
134
135 if (0.4 * double(minCellClusterSize) > totalWeight)
136 return 0;
137
138 // map placement unit to hyperNodes for PaToH
139 int numHyperNodes = 0; // num of placement unit
140 std::vector<int> placementUnit2hyperNode(graphPartitioner->nodeList.size(), -1);
141
142 for (auto instId : inputCluster)
143 placementUnit2hyperNode[instId] = numHyperNodes++;
144
145 int numPlacementNets = 0;
146 std::vector<bool> isCsdNet(graphPartitioner->netList.size(), false);
147 int numPlacementPins = 0;
148 for (auto net : graphPartitioner->netList)
149 {
150 bool exist = false;
151 for (auto tmpPU : net->getUnits())
152 {
153 if (placementUnit2hyperNode[tmpPU->getId()] != -1)
154 {
155 ++numPlacementPins;
156 exist = true;
157 }
158 }
159 if (exist)
160 {
161 ++numPlacementNets;
162 isCsdNet[net->getId()] = true;
163 }
164 }
165
166 unsigned int shareMemorySize = 6 + numHyperNodes + numPlacementNets + 1 + numPlacementPins + numHyperNodes + 2;
167 ExternalProcessFunc *externalProc =
168 new ExternalProcessFunc(getExePath() + "/partitionHyperGraph", shareMemorySize * 4, graphPartitioner->verbose);
169 int *shareMemory = (int *)externalProc->getSharedMemory();
170 shareMemory[0] = numHyperNodes;
171 shareMemory[1] = numPlacementNets;
172 shareMemory[2] = numPlacementPins;
173 double final_imbal = std::max(0.5 - double(minCellClusterSize) / totalWeight, 0.4);
174 *(double *)(shareMemory + 3) = final_imbal;
175 int *cut = shareMemory + 5;
176 int *cwghts = shareMemory + 6;
177 int *xpins = cwghts + numHyperNodes;
178 int *pins = xpins + numPlacementNets + 1;
179 int *partvec = pins + numPlacementPins;
180 int *partweights = partvec + numHyperNodes;
181
182 int *hyperNode2placementUnit = new int[numHyperNodes];
183 for (auto instId : inputCluster)
184 hyperNode2placementUnit[placementUnit2hyperNode[instId]] = instId;
185
186 for (int i = 0; i < numHyperNodes; ++i)
187 {
188 assert((unsigned int)hyperNode2placementUnit[i] < graphPartitioner->nodeList.size());
189 cwghts[i] = graphPartitioner->nodeList[hyperNode2placementUnit[i]]->getWeight();
190 partvec[i] = -1;
191 }
192
193 // map placement nets to hypernet for PaToH
194
195 int maxCut = maxCutRate * numPlacementNets;
196
197 // hyper net to hyper nodes
198 xpins[0] = 0;
199 int indexNet = 0, indexPin = 0;
200 for (auto net : graphPartitioner->netList)
201 {
202 if (isCsdNet[net->getId()])
203 {
204 for (auto tmpPU : net->getUnits())
205 if (placementUnit2hyperNode[tmpPU->getId()] != -1)
206 pins[indexPin++] = placementUnit2hyperNode[tmpPU->getId()];
207 xpins[++indexNet] = indexPin;
208 }
209 }
210 assert(indexPin == numPlacementPins && indexNet == numPlacementNets);
211
212 // call external process to execute
213 externalProc->execute();
214
215 // Postprocessing
216 for (int i = 0; i < numHyperNodes; ++i)
217 outputClusters[partvec[i]].push_back(inputCluster[i]);
218
219 int totalWeight0 = 0;
220 for (auto PUId : outputClusters[0])
221 totalWeight0 += graphPartitioner->nodeList[PUId]->getWeight();
222 int totalWeight1 = 0;
223 for (auto PUId : outputClusters[1])
224 totalWeight1 += graphPartitioner->nodeList[PUId]->getWeight();
225
226 unsigned minClusterTotalWeight = std::min(totalWeight0, totalWeight1);
227 double imbal = 0.5 - double(minClusterTotalWeight) / totalWeight;
228
229 graphPartitioner->cntLock.lock();
230 if ((*cut > maxCut || ((float)outputClusters[0].size() / (float)outputClusters[1].size() < 0.666) ||
231 ((float)outputClusters[1].size() / (float)outputClusters[0].size() < 0.666)) &&
232 (minCellClusterSize * 1.25 > totalWeight && totalDSPNum <= 3 * eachClusterDSPNum &&
233 totalBRAMNum <= 3 * eachClusterBRAMNum))
234 {
235 // if (graphPartitioner->verbose)
236 print_warning("partitioned iter#" + std::to_string(cnt) + " #node: " + std::to_string(numHyperNodes) +
237 " #net: " + std::to_string(numPlacementNets) + " max_cut: " + std::to_string(maxCut) +
238 " max_imbal: " + std::to_string(final_imbal) + " get too large cut: " + std::to_string(*cut) +
239 " failed to find cut meeting requirements of min-cut or size balance. cluster[0].size=" +
240 std::to_string(outputClusters[0].size()) +
241 " cluster[1].size=" + std::to_string(outputClusters[1].size()));
242 outputClusters[0].clear();
243 outputClusters[1].clear();
244 }
245 else
246 {
247 if (graphPartitioner->verbose)
248 print_info("partitioned iter#" + std::to_string(cnt) + " #node: " + std::to_string(numHyperNodes) +
249 " #net: " + std::to_string(numPlacementNets) + " max_cut: " + std::to_string(maxCut) +
250 " max_imbal: " + std::to_string(final_imbal) + " succeed with " +
251 std::to_string(partweights[0]) + " and " + std::to_string(partweights[1]) +
252 " (cut: " + std::to_string(*cut) + ", imbal: " + std::to_string(imbal) + ")" +
253 "cluster[0].size=" + std::to_string(outputClusters[0].size()) +
254 " cluster[1].size=" + std::to_string(outputClusters[1].size()));
255 }
256
257 ++cnt;
258 graphPartitioner->cntLock.unlock();
259
260 int res = *cut;
261 delete[] hyperNode2placementUnit;
262 delete externalProc;
263
264 return res;
265}
