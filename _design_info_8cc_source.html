28DesignInfo::DesignPinType DesignInfo::DesignPin::checkPinType(DesignCell *cell, std::string &_refpinName, bool isInput)
29{
30 if (cell->isLUT())
31 {
32 if (isInput)
33 return PinType_LUTInput;
34 else
35 return PinType_LUTOutput;
36 }
37 else if (cell->isFF())
38 {
39 if (isInput)
40 {
41 if (_refpinName == "R" || _refpinName == "S" || _refpinName == "SRI" || _refpinName == "CLR" ||
42 _refpinName == "PRE")
43 return PinType_SR;
44 if (_refpinName == "C" || _refpinName == "G")
45 return PinType_CLK;
46 if (_refpinName == "D" || _refpinName == "DI")
47 return PinType_D;
48 if (_refpinName == "CE" || _refpinName == "GE")
49 return PinType_E;
50 print_error("no such pin type:" + _refpinName);
51 assert(false && "no such pin type");
52 return PinType_SR; // no meaning
53 }
54 else
55 {
56 if (_refpinName == "Q" || _refpinName == "O")
57 return PinType_Q;
58 std::cout << "cell: " << cell << "\n";
59 print_error("no such pin type:" + _refpinName);
60 assert(false && "no such pin type");
61 return PinType_SR; // no meaning
62 }
63 }
64 else
65 {
66 return PinType_Others;
67 }
