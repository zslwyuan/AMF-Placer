111 inline float getDelayByModel_conservative(float X1, float Y1, float X2, float Y2)
112 {
113 int clockRegionX0, clockRegionY0;
114 deviceInfo->getClockRegionByLocation(X1, Y1, clockRegionX0, clockRegionY0);
115 int clockRegionX1, clockRegionY1;
116 deviceInfo->getClockRegionByLocation(X2, Y2, clockRegionX1, clockRegionY1);
117
118 float X = std::fabs(X1 - X2);
119 float Y = std::fabs(Y1 - Y2);
120
121 if (X * X + Y * Y < 9)
122 {
123 X *= 2;
124 float delay = (timingC0[0] + std::pow(X, 0.3) * timingC0[1] + std::pow(Y, 0.3) * timingC0[2] +
125 std::pow(X, 0.5) * timingC0[3] + std::pow(Y, 0.5) * timingC0[4]) /
126 1000.0;
127 if (std::abs(clockRegionX1 - clockRegionX0) > 1 || clockRegionX1 == 2 || clockRegionX0 == 2)
128 delay += std::abs(clockRegionX1 - clockRegionX0) * 0.5;
129
130 if (delay < 0.05)
131 delay = 0.05;
132 return delay;
133 }
134 else if (X * X + Y * Y < 36)
135 {
136 X *= 2;
137 float delay = (timingC1[0] + std::pow(X, 0.3) * timingC1[1] + std::pow(Y, 0.3) * timingC1[2] +
138 std::pow(X, 0.5) * timingC1[3] + std::pow(Y, 0.5) * timingC1[4]) /
139 1000.0;
140 if (std::abs(clockRegionX1 - clockRegionX0) > 1 || clockRegionX1 == 2 || clockRegionX0 == 2)
141 delay += std::abs(clockRegionX1 - clockRegionX0) * 0.5;
142
143 if (delay < 0.05)
144 delay = 0.05;
145 return delay;
146 }
147 else
148 {
149 X *= 2;
150 float delay = (timingC2[0] + std::pow(X, 0.3) * timingC2[1] + std::pow(Y, 0.3) * timingC2[2] +
151 std::pow(X, 0.5) * timingC2[3] + std::pow(Y, 0.5) * timingC2[4]) /
152 1000.0;
153 if (std::abs(clockRegionX1 - clockRegionX0) > 1 || clockRegionX1 == 2 || clockRegionX0 == 2)
154 delay += std::abs(clockRegionX1 - clockRegionX0) * 0.5;
155

157 delay = 0.05;
