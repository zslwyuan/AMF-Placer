687  std::vector<double> works_E;

698 
699  printProgress(1 - ((double)restartI / restartNum));
700 
701  threads.clear();
702  workers_cluster2XY.clear();
703  workers_grid2clusters.clear();
704  works_E.clear();
705  workers_randomSeed.clear();
706 
707  for (int threadId = restartI; threadId >= 0 && threadId > restartI - nJobs; threadId--)
708  {
709 
710  // generate initial cluster placement
711  init_cluster2XY.clear();
712  init_grid2clusters = std::vector<std::vector<std::vector<int>>>(
713  gridH, std::vector<std::vector<int>>(gridW, std::vector<int>()));
714 
715  initOffset++;

717 
718  for (unsigned int clusterA = 0; clusterA < clusterAdjMat.size(); clusterA++)
719  {
720  assert(init_cluster2XY[clusterA].first >= 0 && init_cluster2XY[clusterA].second >= 0);
721  }
722  for (int gridY = 0; gridY < gridH; gridY++)
723  {
724  for (int gridX = 0; gridX < gridW; gridX++)
725  {
726  std::sort(init_grid2clusters[gridY][gridX].begin(), init_grid2clusters[gridY][gridX].end());
727  }
728  }
729  SACalibrationOffset = evaluateClusterPlacement(init_grid2clusters, init_cluster2XY);
730 
731  workers_cluster2XY.push_back(init_cluster2XY);
732  workers_grid2clusters.push_back(init_grid2clusters);
733  works_E.push_back(0);
734  workers_randomSeed.push_back(threadId);
735  }
736  for (unsigned int threadId = 0; threadId < workers_cluster2XY.size(); threadId++)
737  {
738  threads.push_back(std::thread(worker, this, std::ref(init_grid2clusters), std::ref(init_cluster2XY),
739  std::ref(workers_grid2clusters[threadId]),
740  std::ref(workers_cluster2XY[threadId]), std::ref(workerIterNum),
741  std::ref(workers_randomSeed[threadId]), std::ref(works_E[threadId])));
742  }
743  for (unsigned int threadId = 0; threadId < threads.size(); threadId++)
744  {
745  threads[threadId].join();
746  seedAndE.emplace_back(workers_randomSeed[threadId], works_E[threadId]);
747  }
748

751  resE = works_E[0];

761  res_grid2clusters = workers_grid2clusters[threadId];
762  res_cluster2XY = workers_cluster2XY[threadId];
763  }
764  }
765  }
766  printf("\n");

768  print_info("SA optimization initE = " + std::to_string(SACalibrationOffset));

781  {
782  occupiedRegionNum++;
783  }
784  }
785  print_info("SA final occupied region num = " + std::to_string(occupiedRegionNum));
786 
787  if (dumpSAFile != "")
788  outfile0.close();
789  print_info("SAPlace handle " + std::to_string(init_cluster2XY.size()) +
790  " cluster(s) and the final placement cost is " + std::to_string(resE));
791 }
