20double SAPlacer::evaluateClusterPlacement(const std::vector<std::vector<std::vector<int>>> &grid2clusters,
21 const std::vector<std::pair<int, int>> &cluster2XY)
22{
23
24 float regionW = deviceW / gridW;
25 float regionH = deviceH / gridH;
26
27 double resWL = 0;
28
29 for (unsigned int clusterA = 1; clusterA < clusterAdjMat.size(); clusterA++)
30 for (unsigned int clusterB = 0; clusterB < clusterA; clusterB++)
31 {
32 if (clusterAdjMat[clusterA][clusterB] > 0.00001)
33 {
34 resWL += clusterAdjMat[clusterA][clusterB] *
35 (fabs(cluster2XY[clusterA].first - cluster2XY[clusterB].first) * regionW +
36 y2xRatio * fabs(cluster2XY[clusterA].second - cluster2XY[clusterB].second) * regionH);
37 if (cluster2XY[clusterA].first == 2 || cluster2XY[clusterB].first == 2)
38 resWL += fabs(cluster2XY[clusterA].first - cluster2XY[clusterB].first) * regionW * 0.5;
39 }
40 }
41 for (unsigned int clusterA = 0; clusterA < clusterAdjMat.size(); clusterA++)
42 {
43 for (unsigned int fixedUnitId = 0; fixedUnitId < cluster2FixedUnitMat[clusterA].size(); fixedUnitId++)
44 {
45 if (cluster2FixedUnitMat[clusterA][fixedUnitId] > 0.00001)
46 {
47 resWL += connectionToFixedFactor * cluster2FixedUnitMat[clusterA][fixedUnitId] *
48 (fabs((cluster2XY[clusterA].first * regionW + regionW / 2) - fixedX[fixedUnitId]) +
49 y2xRatio * fabs((cluster2XY[clusterA].second * regionH + regionH / 2) - fixedY[fixedUnitId]));
50 }
51 }
52 }
53
54 for (int gridY = 0; gridY < gridH; gridY++)
55 for (int gridX = 0; gridX < gridW; gridX++)
56 {
57 if (grid2clusters[gridY][gridX].size() > 1)
58 {
59 // resWL *= 10;
60 for (unsigned int clusterAId = 1; clusterAId < grid2clusters[gridY][gridX].size(); clusterAId++)
61 for (unsigned int clusterBId = 0; clusterBId < clusterAId; clusterBId++)
62 {
63 int clusterA = grid2clusters[gridY][gridX][clusterAId];
64 int clusterB = grid2clusters[gridY][gridX][clusterBId];
65 resWL +=
66 5000 *
67 std::pow(std::min(1.1, (clusterWeights[clusterA] + clusterWeights[clusterB]) / 20000.0),
68 2) *
69 (regionW + y2xRatio * regionH);
70 }
71 }
72 }
73
74 return resWL;

423float SAPlacer::probabilituFunc(double oriE, double newE, float T)
424{
425 if (newE < oriE)
