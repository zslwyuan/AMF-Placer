455 inline unsigned int getPairPinNum(DesignInfo::DesignCell *LUTA, DesignInfo::DesignCell *LUTB)
456 {
457 if (LUTA->getInputPins().size() == 6 || LUTB->getInputPins().size() == 6 || LUTA->isLUT6() ||
458 LUTB->isLUT6())
459 return 12;
460
461 int pinNumA = 0;
462 int totalPin = 0;
463 int netIds[5]; // be aware that a LUT might have pins connected to the same net and they should be
464 // treated as different inputs.
465
466 for (auto tmpPin : LUTA->getInputPins())
467 {
468 if (!tmpPin->isUnconnected())
469 {
470 netIds[pinNumA] = tmpPin->getNet()->getElementIdInType();
471 pinNumA++;
472 }
473 }
474 totalPin = pinNumA;
475 for (auto tmpPin : LUTB->getInputPins())
476 {
477 if (!tmpPin->isUnconnected())
478 {
479 bool matched = false;
480 for (int i = 0; i < pinNumA; i++)
481 {
482 if (netIds[i] >= 0 && netIds[i] == tmpPin->getNet()->getElementIdInType())
483 {
484 netIds[i] = -1;
485 matched = true;
486 break;
487 }
488 }
489 if (!matched)
490 {
491 totalPin++;
492 }
493 }
494 }
495
496 return totalPin;
