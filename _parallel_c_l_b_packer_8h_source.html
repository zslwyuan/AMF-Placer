510 inline unsigned int getPairPinNum(DesignInfo::DesignCell *LUTA, DesignInfo::DesignCell *LUTB)
511 {
512 if (LUTA->getInputPins().size() == 6 || LUTB->getInputPins().size() == 6 || LUTA->isLUT6() ||
513 LUTB->isLUT6())
514 return 12;
515
516 int pinNumA = 0;
517 int totalPin = 0;
518 int netIds[5]; // be aware that a LUT might have pins connected to the same net and they should be
519 // treated as different inputs.
520
521 for (auto tmpPin : LUTA->getInputPins())
522 {
523 if (!tmpPin->isUnconnected())
524 {
525 netIds[pinNumA] = tmpPin->getNet()->getElementIdInType();
526 pinNumA++;
527 }
528 }
529 totalPin = pinNumA;
530 for (auto tmpPin : LUTB->getInputPins())
531 {
532 if (!tmpPin->isUnconnected())
533 {
534 bool matched = false;
535 for (int i = 0; i < pinNumA; i++)
536 {
537 if (netIds[i] >= 0 && netIds[i] == tmpPin->getNet()->getElementIdInType())
538 {
539 netIds[i] = -1;
540 matched = true;
541 break;
542 }
543 }
544 if (!matched)
545 {
546 totalPin++;
547 }
548 }
549 }
550
551 return totalPin;
552 }
