458 inline unsigned int getPairPinNum(DesignInfo::DesignCell *LUTA, DesignInfo::DesignCell *LUTB)
459 {
460 if (LUTA->getInputPins().size() == 6 || LUTB->getInputPins().size() == 6 || LUTA->isLUT6() ||
461 LUTB->isLUT6())
462 return 12;
463
464 int pinNumA = 0;
465 int totalPin = 0;
466 int netIds[5]; // be aware that a LUT might have pins connected to the same net and they should be
467 // treated as different inputs.
468
469 for (auto tmpPin : LUTA->getInputPins())
470 {
471 if (!tmpPin->isUnconnected())
472 {
473 netIds[pinNumA] = tmpPin->getNet()->getElementIdInType();
474 pinNumA++;
475 }
476 }
477 totalPin = pinNumA;
478 for (auto tmpPin : LUTB->getInputPins())
479 {
480 if (!tmpPin->isUnconnected())
481 {
482 bool matched = false;
483 for (int i = 0; i < pinNumA; i++)
484 {
485 if (netIds[i] >= 0 && netIds[i] == tmpPin->getNet()->getElementIdInType())
486 {
487 netIds[i] = -1;
488 matched = true;
489 break;
490 }
491 }
492 if (!matched)
493 {
494 totalPin++;
495 }
496 }
497 }
498
499 return totalPin;
