711 inline float getHPWLChange(DesignInfo::DesignCell *curCell, DeviceInfo::DeviceSite *curSite)
712 {
713 float oriHPWL = 0.0;
714 float newHPWL = 0.0;
715 auto tmpPU = placementInfo->getPlacementUnitByCell(curCell);
716 float PUX = 0.0, PUY = 0.0;
717 auto nets = placementInfo->getPlacementUnitId2Nets()[tmpPU->getId()];
718 float numCellsInMacro = 1.0;
719 if (dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(tmpPU))
720 {
721 PUX = curSite->X();
722 PUY = curSite->Y();
723 }
724 else if (PlacementInfo::PlacementMacro *tmpMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(tmpPU))
725 {
726
727 PUX = curSite->X() - tmpMacro->getCellOffsetXInMacro(curCell);
728 PUY = curSite->Y() - tmpMacro->getCellOffsetYInMacro(curCell);
729 numCellsInMacro = tmpMacro->getCells().size();
730 }
731
732 for (auto curNet : nets)
733 {
734 if (curNet->getDesignNet()->getPins().size() > 1000) // it could be clock
735 continue;
736 oriHPWL += curNet->getHPWL(y2xRatio);
737 newHPWL += curNet->getNewHPWLByTrying(tmpPU, PUX, PUY, y2xRatio);
738 }
739 return (newHPWL - oriHPWL) / numCellsInMacro;
740 // return std::fabs(macroLoc.X - curSite->X()) + std::fabs(macroLoc.Y - curSite->Y());
741 // placementInfo->getPlacementUnitByCell(curCell);
742 }

805 inline int sortPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
806 int high, PlacementInfo::Location &macroLoc)
807 {
808 int pivot, index, i;
809 index = low;
810 pivot = high;
811 for (i = low; i < high; i++)
812 {
813 // finding index of pivot.
814 // if (a[i] < a[pivot])
815 // if (getDisplacement(macroLoc, sites[i]) < getDisplacement(macroLoc, sites[pivot]))
816 if (getHPWLChange(curCell, sites[i]) < getHPWLChange(curCell, sites[pivot]))
817 {
818 swapSitePtr(&sites[i], &sites[index]);
819 index++;
820 }
821 }
822 swapSitePtr(&sites[pivot], &sites[index]);
823 return index;
824 }
825 inline int RandomPivotPartition(DesignInfo::DesignCell *curCell, std::vector<DeviceInfo::DeviceSite *> &sites,
826 int low, int high, PlacementInfo::Location &macroLoc)
827 {
828 // Random selection of pivot.
829 int pvt;
830 pvt = (high + low) / 2;
831 // pvt = low + n % (high - low + 1); // Randomizing the pivot value from sub-array.
832 swapSitePtr(&sites[high], &sites[pvt]);
833 return sortPartition(curCell, sites, low, high, macroLoc);
834 }
