20bool siteSortCmp(DeviceInfo::DeviceSite *a, DeviceInfo::DeviceSite *b)
21{
22 const float eps = 1e-3;
23 if (a->Y() <= b->Y() + eps && a->Y() >= b->Y() - eps)
24 return a->X() < b->X();
25 else if (a->Y() < b->Y() - eps)
26 return true;
27 else if (a->Y() > b->Y() + eps)
28 return false;
29 return true;
30}

160{

163 for (int i = 0; i < clockRegionNumY; i++)

183 {
184 int i = 0;

191 }

194 {
195 for (int j = 0; j < clockRegionNumX; j++)
196 {
197 if (j > 0)
198 {

200 clockRegions[i][j]->setLeft(midX);
201 clockRegions[i][j - 1]->setRight(midX);
202 }

234void DeviceInfo::ClockRegion::mapSiteToClockColumns()
235{
236 assert(sites.size() > 0);
237 std::sort(sites.begin(), sites.end(), [](DeviceSite *a, DeviceSite *b) -> bool {
238 return a->getSiteY() == b->getSiteY() ? (a->getSiteX() < b->getSiteX()) : a->getSiteY() < b->getSiteY();
239 });
240
241 assert(sites.size() > 0);
242 for (unsigned int i = 0; i < sites.size(); i++)
243 {
244 if (sites[i]->getName().find("SLICE") != std::string::npos)
245 {
246 leftSiteX = sites[i]->getSiteX();
247 bottomSiteY = sites[i]->getSiteY();
248 break;
249 }
250 }
251 for (int i = sites.size() - 1; i >= 0; i--)
252 {
253 if (sites[i]->getName().find("SLICE") != std::string::npos)
254 {
255 rightSiteX = sites[i]->getSiteX();
256 topSiteY = sites[i]->getSiteY();
257 break;
258 }
259 }
260
261 clockColumns = std::vector<std::vector<ClockColumn>>(
262 columnNumY, std::vector<ClockColumn>(rightSiteX - leftSiteX + 1, ClockColumn()));
263
264 int eachLevelY = (topSiteY - bottomSiteY + 1) / columnNumY;
265 for (auto curSite : sites)
266 {
267 if (curSite->getName().find("SLICE") == std::string::npos)
268 {
269 // currently, BRAM/DSP slices will not lead to clock utilization overflow
270 continue;
271 }
272 int levelY = (curSite->getSiteY() - bottomSiteY) / eachLevelY;
273 assert(levelY >= 0);
274 assert(levelY < (int)clockColumns.size());
275 assert(curSite->getSiteX() - leftSiteX >= 0);
276 assert(curSite->getSiteX() - leftSiteX < (int)clockColumns[levelY].size());
277 clockColumns[levelY][curSite->getSiteX() - leftSiteX].addSite(curSite);
278 }
279
280 assert(clockColumns.size() > 0);
281 colHeight = (topY - bottomY) / columnNumY;
282 colWidth = (rightX - leftX) / clockColumns[0].size();
283
284 float offsetY = bottomY;
285 for (int levelY = 0; levelY < columnNumY; levelY++)
286 {
287 float offsetX = leftX;
288 for (unsigned int colOffset = 0; colOffset < clockColumns[levelY].size(); colOffset++)
289 {
290 clockColumns[levelY][colOffset].setBoundary(offsetX, offsetX + colWidth, offsetY + colHeight, offsetY);
291 offsetX += colWidth;
292 }
293 offsetY += colHeight;
294 }
