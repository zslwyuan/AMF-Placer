20bool siteSortCmp(DeviceInfo::DeviceSite *a, DeviceInfo::DeviceSite *b)
21{
22 const float eps = 1e-3;
23 if (a->Y() <= b->Y() + eps && a->Y() >= b->Y() - eps)
24 return a->X() < b->X();
25 else if (a->Y() < b->Y() - eps)
26 return true;
27 else if (a->Y() > b->Y() + eps)
28 return false;
29 return true;
30}

252void DeviceInfo::ClockRegion::mapSiteToClockColumns()
253{
254 assert(sites.size() > 0);
255 std::sort(sites.begin(), sites.end(), [](DeviceSite *a, DeviceSite *b) -> bool {
256 return a->getSiteY() == b->getSiteY() ? (a->getSiteX() < b->getSiteX()) : a->getSiteY() < b->getSiteY();
257 });
258
259 assert(sites.size() > 0);
260 for (unsigned int i = 0; i < sites.size(); i++)
261 {
262 if (sites[i]->getName().find("SLICE") == 0)
263 {
264 leftSiteX = sites[i]->getSiteX();
265 bottomSiteY = sites[i]->getSiteY();
266 break;
267 }
268 }
269 for (int i = sites.size() - 1; i >= 0; i--)
270 {
271 if (sites[i]->getName().find("SLICE") == 0)
272 {
273 rightSiteX = sites[i]->getSiteX();
274 topSiteY = sites[i]->getSiteY();
275 break;
276 }
277 }
278
279 for (unsigned int i = 0; i < sites.size(); i++)
280 {
281 if (sites[i]->getTile()->getTileIdX() < leftTileIdX)
282 {
283 leftTileIdX = sites[i]->getTile()->getTileIdX();
284 }
285 if (sites[i]->getTile()->getTileIdX() > rightTileIdX)
286 {
287 rightTileIdX = sites[i]->getTile()->getTileIdX();
288 }
289 if (sites[i]->getName().find("SLICE") == 0)
290 {
291 if (sites[i]->getTile()->getTileIdY() < bottomTileIdY)
292 {
293 bottomTileIdY = sites[i]->getTile()->getTileIdY();
294 }
295 if (sites[i]->getTile()->getTileIdY() > topTileIdY)
296 {
297 topTileIdY = sites[i]->getTile()->getTileIdY();
298 }
299 }
300 }
301
302 // std::cout << "<<<<<<<<<<<<<<<<<<<<<<<<<<<<\n";
303 clockColumns = std::vector<std::vector<ClockColumn *>>(
304 columnNumY, std::vector<ClockColumn *>((rightTileIdX - leftTileIdX + 1), nullptr));
305 for (int levelY = 0; levelY < columnNumY; levelY++)
306 {
307 for (unsigned int colOffset = 0; colOffset < clockColumns[levelY].size(); colOffset++)
308 {
309 clockColumns[levelY][colOffset] = new ClockColumn();
310 }
311 }
312 int eachLevelY = (topTileIdY - bottomTileIdY + 1) / columnNumY;
313 for (auto curSite : sites)
314 {
315 if (curSite->getName().find("SLICE") != 0)
316 {
317 // currently, BRAM/DSP slices will not lead to clock utilization overflow
318 continue;
319 }
320 int levelY = (curSite->getSiteY() - bottomSiteY) / eachLevelY;
321 int offsetX = curSite->getTile()->getTileIdX() - leftTileIdX;
322 if (levelY < 0)
323 {
324 std::cout << curSite->getName() << "\n";
325 }
326 assert(levelY >= 0);
327 if (levelY >= (int)clockColumns.size())
328 {
329 std::cout << curSite->getName() << "\n";
330 }
331 assert(levelY < (int)clockColumns.size());
332 assert(offsetX >= 0);
333 assert(offsetX < (int)clockColumns[levelY].size());
334 clockColumns[levelY][offsetX]->addSite(curSite);
335 curSite->setClockHalfColumn(clockColumns[levelY][offsetX]);
336 }
337
338 assert(clockColumns.size() > 0);
339 colHeight = (topY - bottomY) / columnNumY;
340 colWidth = (rightX - leftX) / clockColumns[0].size();
341
342 float offsetY = bottomY;
343 // std::cout << "ClockRegionBoundary: l:" << leftX << " r:" << rightX << " b:" << bottomY << " t:" << topY << "\n";
344 for (int levelY = 0; levelY < columnNumY; levelY++)
345 {
346 for (unsigned int colOffset = 0; colOffset < clockColumns[levelY].size(); colOffset++)
347 {
348 if (clockColumns[levelY][colOffset]->getSites().size() == 0)
349 continue;
350 clockColumns[levelY][colOffset]->setBottom(bottomY + levelY * colHeight);
351 clockColumns[levelY][colOffset]->setTop(bottomY + (levelY + 1) * colHeight);
352 if (colOffset == 0)
353 {
354 clockColumns[levelY][colOffset]->setLeft(leftX);
355 }
356 else
357 {
358 clockColumns[levelY][colOffset]->setLeft(clockColumns[levelY][colOffset]->getLeft() - 0.25);
359 }
360 if (colOffset == clockColumns[levelY].size() - 1)
361 {
362 clockColumns[levelY][colOffset]->setRight(rightX);
363 }
364 else
365 {
366 clockColumns[levelY][colOffset]->setRight(clockColumns[levelY][colOffset]->getRight() + 0.25);
367 }
368 // std::cout << " l:" << clockColumns[levelY][colOffset]->getLeft()
369 // << " r:" << clockColumns[levelY][colOffset]->getRight()
370 // << " b:" << clockColumns[levelY][colOffset]->getBottom()
371 // << " t:" << clockColumns[levelY][colOffset]->getTop() << "\n";
372 }
373 }
