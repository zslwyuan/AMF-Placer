169float CLBLegalizer::DPForMinHPWL(int colNum, std::vector<std::vector<DeviceInfo::DeviceSite *>> &Column2Sites,
170 std::vector<std::deque<PlacementInfo::PlacementUnit *>> &Column2PUs)
171{
172 // final Legalization DP
173 // i th macro (start from 0), j th row (start from 0)
174 // f[i][j] = min(f[i-1][j-row[i]]+HPWLChange[i][j-row[i]+1],f[i][j-1])
175
176 // std::string dumpFile =
177 // JSONCfg["DumpCLBLegalization"] + "-Exact-" + std::to_string(DumpCLBLegalizationCnt) + ".gz";
178
179 // print_status("CLBLegalizer: dumping CLBLegalization archieve to: " + dumpFile);
180 // DumpCLBLegalizationCnt++;
181 // std::stringstream outfile0;
182
183 float tmpTotalDisplacement = 0.0;
184 std::map<PlacementInfo::PlacementUnit *, std::vector<DeviceInfo::DeviceSite *>> PU2Sites;
185 PU2Sites.clear();
186 for (auto &PUDeque : Column2PUs)
187 {
188 for (auto tmpPU : PUDeque)
189 {
190 assert(PU2Sites.find(tmpPU) == PU2Sites.end());
191 PU2Sites[tmpPU] = std::vector<DeviceInfo::DeviceSite *>();
192 }
193 }
194
195#pragma omp parallel for
196 for (int c = 0; c < colNum; c++)
197 {
198 int numPUs = Column2PUs[c].size();
199 if (!numPUs)
200 continue;
201
202 auto &curColSites = Column2Sites[c];
203 auto &curColPU = Column2PUs[c];
204
205 sortSitesBySiteY(curColSites);
206
207 // initialize
208
209 std::vector<std::vector<float>> f(numPUs, std::vector<float>(curColSites.size(), 1000000000.0));
210 std::vector<std::vector<bool>> fChoice(
211 numPUs, std::vector<bool>(curColSites.size(), 0)); // used for tracing back macro-site mapping
212
213 int minLastPUTop = -1;
214
215 int totalMacroCellNum = 0;
216 int heightPURow = getPUSiteNum(curColPU[0]);
217 totalMacroCellNum += heightPURow;
218 float minHPWLChange = 1100000000.0;
219 for (unsigned int j = heightPURow - 1; j < curColSites.size(); j++)
220 {
221 float curHPWLChange = getHPWLChange(curColPU[0], curColSites[j - heightPURow + 1]);
222 if (curColSites[j]->getSiteY() - curColSites[j - heightPURow + 1]->getSiteY() != heightPURow - 1)
223 {
224 // we need to ensure that there is no occpupied sites in this range
225 curHPWLChange = 1100000000.0;
226 }
227 if (curHPWLChange < minHPWLChange)
228 {
229 minHPWLChange = curHPWLChange;
230 fChoice[0][j] = 1;
231 if (numPUs == 1)
232 {
233 minLastPUTop = j;
234 }
235 }
236 f[0][j] = minHPWLChange;
237 }
238
239 // DP to minimize HPWL
240 for (int i = 1; i < numPUs; i++)
241 {
242 float minVal = 100000000.0;
243 int heightPURow = getPUSiteNum(curColPU[i]);
244 totalMacroCellNum += heightPURow;
245 for (unsigned int j = totalMacroCellNum - 1; j < curColSites.size();
246 j++) // j start from heightPURow because PU0 must occupy 1+ site(s)
247 {
248 float curHPWLChange = getHPWLChange(curColPU[i], curColSites[j - heightPURow + 1]);
249 if (curColSites[j]->getSiteY() - curColSites[j - heightPURow + 1]->getSiteY() != heightPURow - 1)
250 {
251 // we need to ensure that there is no occpupied sites in this range
252 curHPWLChange = 1100000000.0;
253 }
254 if (f[i][j - 1] > f[i - 1][j - heightPURow] + curHPWLChange)
255 {
256 fChoice[i][j] = 1;
257 if (i == numPUs - 1)
258 {
259 if (f[i - 1][j - heightPURow] + curHPWLChange < minVal)
260 {
261 minVal = f[i - 1][j - heightPURow] + curHPWLChange;
262 minLastPUTop = j;
263 }
264 }
265 }
266
267 f[i][j] = std::min(f[i][j - 1], f[i - 1][j - heightPURow] + curHPWLChange);
268 }
269 }
270
271 assert(minLastPUTop >= 0);
272
273 std::vector<DeviceInfo::DeviceSite *> PUBottom2Site_reverse(0);
274
275 for (int i = numPUs - 1; i >= 0; i--)
276 {
277 assert(minLastPUTop >= 0);
278 int heightPURow = getPUSiteNum(curColPU[i]);
279 while (!fChoice[i][minLastPUTop])
280 {
281 minLastPUTop--;
282 assert(minLastPUTop >= 0);
283 }
284 auto curSite = curColSites[minLastPUTop - heightPURow + 1];
285 assert(PU2Sites[curColPU[i]].size() == 0);
286 assert(curSite);
287 PU2Sites[curColPU[i]].push_back(curSite); // CLB PU will only occupy one site
288 minLastPUTop -= heightPURow;
289 // outfile0 << "col#" << c << ": " << curColPU[i]->getName() << " PUY:" << PU2Y[curColPU[i]]
290 // << " numPUs:" << getPUSiteNum(curColPU[i])
291 // << " netNum:" << curColPU[i]->getNetsSetPtr()->size() << "\n";
292 // outfile0 << " macthed with: " << curSite->getName() << " locX:" << curSite->X()
293 // << " locY:" << curSite->Y() << " HPWLIncrease:" << getHPWLChange(curColPU[i], curSite) << "\n";
294 }
295 }
296
297 for (auto &PUDeque : Column2PUs)
298 {
299 for (auto tmpPU : PUDeque)
300 {
301 assert(PU2Sites.find(tmpPU) != PU2Sites.end());
302 auto curSite = PU2Sites[tmpPU][0];
303 assert(curSite);
304 assert(PU2X.find(tmpPU) == PU2X.end());
305 PU2X[tmpPU] = curSite->X();
306 PU2Y[tmpPU] = curSite->Y();
307 PU2LegalSites[tmpPU] = PU2Sites[tmpPU];
308 PULevelMatching.emplace_back(tmpPU, curSite);
309 tmpTotalDisplacement += std::fabs(tmpPU->X() - curSite->X()) + std::fabs(tmpPU->Y() - curSite->Y());
310 }
311 }
312
313 // writeStrToGZip(dumpFile, outfile0);
314 // print_status("CLBLegalizer: dumped CLBLegalization archieve to: " + dumpFile);
315 return tmpTotalDisplacement;
316}

454void CLBLegalizer::findPossibleLegalLocation(bool fixedColumn)
455{
456 PU2Sites.clear();
457
458 for (auto curPU : PUsToLegalize)
459 {
460 PU2Sites[curPU] = std::vector<DeviceInfo::DeviceSite *>(0);
461 }
462
463 int numPUs = PUsToLegalize.size();
464
465#pragma omp parallel for
466 for (int i = 0; i < numPUs; i++)
467 {
468 auto curPU = PUsToLegalize[i];
469 std::string curSiteType = "";
470 if (curPU->isLCLB())
471 curSiteType = "SLICEL";
472 else if (curPU->isMCLB())
473 curSiteType = "SLICEM";
474 else
475 assert(false && "should be LogicCLB or RAMCLB");
476
477 assert(siteType2Sites.find(curSiteType) != siteType2Sites.end());
478 std::vector<DeviceInfo::DeviceSite *> *candidateSite = nullptr;
479 if (fixedColumn)
480 {
481 int targetSiteX = -1;
482 std::vector<std::vector<DeviceInfo::DeviceSite *>> *column2Sites;
483 if (curPU->isLCLB())
484 {
485 assert(LCLB2Column.find(curPU) != LCLB2Column.end());
486 targetSiteX = LCLB2Column[curPU];
487 column2Sites = &LCLBColumn2Sites;
488 }
489 if (curPU->isMCLB())
490 {
491 assert(MCLB2Column.find(curPU) != MCLB2Column.end());
492 targetSiteX = MCLB2Column[curPU];
493 column2Sites = &MCLBColumn2Sites;
494 }
495 assert(targetSiteX >= 0 && "undefine type");
496
497 for (auto &tmpCol : *column2Sites)
498 {
499 if (tmpCol.size() == 0)
500 continue;
501 if (tmpCol[0]->getSiteX() == targetSiteX)
502 {
503 candidateSite = &tmpCol;
504 }
505 }
506 }
507 else
508 {
509 candidateSite = PU2SitesInDisplacementThreshold[curPU];
510 }
511
512 assert(candidateSite);
513
514 for (auto curSite : *candidateSite)
515 {
516 float curDisp = getDisplacement(curPU, curSite);
517 if ((!fixedColumn && curDisp < displacementThreshold) ||
518 (fixedColumn && curDisp < 2 * displacementThreshold))
519 {
520 if (placementInfo->isLegalLocation(curPU, curSite->X(), curSite->Y()))
521 {
522 if (matchedSites.find(curSite) !=
523 matchedSites.end()) // the matched sites should not in the candidate set
524 continue;
525 PU2Sites[curPU].push_back(curSite);
526 }
527 }
528 }
529 if (PU2Sites[curPU].size() > 1)
530 quick_sort_WLChange(curPU, PU2Sites[curPU], 0, PU2Sites[curPU].size() - 1);
531 if (PU2Sites[curPU].size() > (unsigned int)maxNumCandidate)
532 PU2Sites[curPU].resize(maxNumCandidate);
533 }
534 // print_info("#total macro cell = " + std::to_string(PU2Sites.size()));
535 // print_info("#total macro candidate site (might be duplicated) = " + std::to_string(totalSiteNum));
536}

538void CLBLegalizer::createBipartiteGraph()
539{
540 rightSiteIds.clear();
541 adjList.resize(PUsToLegalize.size());
542 siteList.clear();
543
544 float minCost = 10000000;
545
546 for (unsigned int leftCellId = 0; leftCellId < PUsToLegalize.size(); leftCellId++)
547 {
548 auto curPU = PUsToLegalize[leftCellId];
549 adjList[leftCellId].clear();
550 for (auto curSite : PU2Sites[curPU])
551 {
552 if (rightSiteIds.find(curSite) == rightSiteIds.end())
553 {
554 int curSiteCnt = rightSiteIds.size();
555 rightSiteIds[curSite] = curSiteCnt;
556 siteList.push_back(curSite);
557 }
558 float tmpCost = getHPWLChange(curPU, curSite);
559 adjList[leftCellId].emplace_back(rightSiteIds[curSite], tmpCost);
560 if (tmpCost < minCost)
561 {
562 minCost = tmpCost;
563 }
564 // adjList[leftCellId].emplace_back(rightSiteIds[curSite],
565 // getDisplacement(cellLoc[curCell->getCellId()], curSite));
566 }
567 }
568
569 if (minCost < 0.0001)
570 {
571 float compensation = 10 - minCost;
572 for (unsigned int leftCellId = 0; leftCellId < PUsToLegalize.size(); leftCellId++)
573 {
574 for (unsigned int tmpId = 0; tmpId < adjList[leftCellId].size(); tmpId++)
575 {
576 adjList[leftCellId][tmpId].second += compensation;
577 }
578 }
579 }

582void CLBLegalizer::updateMatchingAndUnmatchedPUs()
583{
584 for (unsigned int leftCellId = 0; leftCellId < PUsToLegalize.size(); leftCellId++)
585 {
586 int rightNode = minCostBipartiteMatcher->getMatchedRightNode(leftCellId);
587 auto curPU = PUsToLegalize[leftCellId];
588 if (rightNode >= 0)
589 {
590 assert(matchedPUs.find(curPU) == matchedPUs.end());
591 assert(matchedSites.find(siteList[rightNode]) == matchedSites.end());
592 matchedPUs.insert(curPU);
593 matchedSites.insert(siteList[rightNode]);
594 PULevelMatching.emplace_back(curPU, siteList[rightNode]);
595 }
596 }
597 std::vector<PlacementInfo::PlacementUnit *> newPUsToLegalize;
598 newPUsToLegalize.clear();
599 for (unsigned int leftCellId = 0; leftCellId < PUsToLegalize.size(); leftCellId++)
600 {
601 auto curPU = PUsToLegalize[leftCellId];
602 if (matchedPUs.find(curPU) == matchedPUs.end())
603 {
604 newPUsToLegalize.push_back(curPU);
605 }
606 }
607 PUsToLegalize = newPUsToLegalize;
608}

701void CLBLegalizer::updatePUMatchingLocation(bool isRoughLegalization, bool updateDisplacement)
702{
703 PU2X.clear();
704 PU2Y.clear();
705 PU2Columns.clear();
706 PU2SiteX.clear();
707
708 PU2LegalSites.clear();
709 MCLBColumn2PUs.clear();
710 if (MCLBColumnNum > 0)
711 {
712 MCLBColumn2PUs.resize(MCLBColumnNum, std::deque<PlacementInfo::PlacementUnit *>(0));
713 }
714 LCLBColumn2PUs.clear();
715 if (LCLBColumnNum > 0)
716 {
717 LCLBColumn2PUs.resize(LCLBColumnNum, std::deque<PlacementInfo::PlacementUnit *>(0));
718 }
719
720 float tmpAverageDisplacement = 0.0;
721
722 for (auto matchedPair : PULevelMatching)
723 {
724 auto matchedSite = matchedPair.second;
725 auto curPU = matchedPair.first;
726
727 if (PU2Y.find(curPU) == PU2Y.end())
728 {
729 PU2SiteX[curPU] = -1;
730 PU2Y[curPU] = 0.0;
731 if (curPU->isMCLB())
732 {
733 MCLBColumn2PUs[matchedSite->getSiteX()].push_back(curPU);
734 }
735 if (curPU->isLCLB())
736 {
737 LCLBColumn2PUs[matchedSite->getSiteX()].push_back(curPU);
738 }
739 }
740
741 float actualPUX = 0;
742 float actualPUY = 0;
743 // since the matched site is for a single DSP/BRAM cell, we need to find its corresponding placement macro
744 // (with multiple DSPs/BRAMs)
745 actualPUX = matchedSite->X();
746 actualPUY = matchedSite->Y();
747
748 tmpAverageDisplacement += std::fabs(curPU->X() - matchedSite->X()) + std::fabs(curPU->Y() - matchedSite->Y());
749 assert(isRoughLegalization || PU2SiteX[curPU] == -1 || PU2SiteX[curPU] == matchedSite->getSiteX());
750 PU2SiteX[curPU] = matchedSite->getSiteX();
751 PU2X[curPU] += actualPUX;
752 PU2Y[curPU] += actualPUY;
753 if (PU2Columns.find(curPU) == PU2Columns.end())
754 {
755 PU2Columns[curPU] = std::vector<int>(0);
756 }
757 PU2Columns[curPU].push_back(matchedSite->getSiteX());
758 }
759
760 tmpAverageDisplacement /= PULevelMatching.size();
761
762 if (verbose)
763 print_info("CLBLegalizer Macro Cell Average Displacement = " + std::to_string(tmpAverageDisplacement));
764
765 if (updateDisplacement)
766 {
767 if (isRoughLegalization)
768 roughAverageDisplacement = tmpAverageDisplacement;
769 else
770 fixedColumnAverageDisplacement = tmpAverageDisplacement;
771 }
772
773 for (auto pairPU2X : PU2X)
774 {
775 int numCells = getPUSiteNum(pairPU2X.first);
776 assert(numCells == 1); // currentlt CLB should only cover one site.
777 PU2X[pairPU2X.first] /= numCells;
778 PU2Y[pairPU2X.first] /= numCells;
779 }
780
781#pragma omp parallel for
782 for (int i = 0; i < MCLBColumnNum; i++)
783 sortPUsByPU2Y(MCLBColumn2PUs[i]);
784
785#pragma omp parallel for
786 for (int i = 0; i < LCLBColumnNum; i++)
787 sortPUsByPU2Y(LCLBColumn2PUs[i]);
788
789 placementInfo->setPULegalXY(PU2X, PU2Y);
790}

792void CLBLegalizer::spreadPUs(int columnNum, std::vector<int> &columnUntilization,
793 std::vector<std::vector<DeviceInfo::DeviceSite *>> &column2Sites,
794 std::vector<std::deque<PlacementInfo::PlacementUnit *>> &column2PUs,
795 std::map<PlacementInfo::PlacementUnit *, int> &cell2Column)
796{
797 while (true)
798 {
799 int overflowColId = -1;
800 std::vector<int> accumulationUtil(columnNum, 0), accumulationCapacity(columnNum, 0);
801 accumulationUtil[0] = columnUntilization[0];
802 accumulationCapacity[0] = column2Sites[0].size();
803 for (int colId = 1; colId < columnNum; colId++)
804 {
805 accumulationUtil[colId] = accumulationUtil[colId - 1] + columnUntilization[colId];
806 accumulationCapacity[colId] = accumulationCapacity[colId - 1] + column2Sites[colId].size();
807 }
808
809 for (int colId = 0; colId < columnNum; colId++)
810 {
811 if (column2Sites[colId].size() < (unsigned int)columnUntilization[colId])
812 {
813 overflowColId = colId;
814 break;
815 }
816 }
817 if (overflowColId < 0)
818 {
819 break;
820 }
821 else
822 {
823 int leftAvaliableCapacity = 0;
824 int rightAvaliableCapacity = 0;
825 int leftUtil = 0;
826 int rightUtil = 0;
827
828 if (overflowColId > 0)
829 {
830 leftAvaliableCapacity += accumulationCapacity[overflowColId - 1];
831 leftUtil += accumulationUtil[overflowColId - 1];
832 }
833 if (overflowColId < columnNum - 1)
834 {
835 rightAvaliableCapacity += accumulationCapacity[columnNum - 1] - accumulationCapacity[overflowColId];
836 rightUtil += accumulationUtil[columnNum - 1] - accumulationUtil[overflowColId];
837 }
838
839 int overflowNum = (unsigned int)columnUntilization[overflowColId] - column2Sites[overflowColId].size();
840 int toLeft = 0;
841
842 int totalAvailableCapacity = rightAvaliableCapacity + leftAvaliableCapacity - leftUtil - rightUtil;
843 assert(totalAvailableCapacity > 0);
844 float toLeftRatio = (float)(leftAvaliableCapacity - leftUtil) / totalAvailableCapacity;
845 int toLeftNum = int((overflowNum * toLeftRatio) + 0.4999);
846 int toRightNum = overflowNum - toLeft;
847
848 if (leftAvaliableCapacity - leftUtil > 0)
849 {
850 while (toLeftNum > 0 && column2PUs[overflowColId].size() > 0)
851 {
852 column2PUs[overflowColId - 1].push_back(column2PUs[overflowColId][0]);
853 int macroSize = getPUSiteNum(column2PUs[overflowColId][0]);
854 columnUntilization[overflowColId - 1] += macroSize;
855 columnUntilization[overflowColId] -= macroSize;
856 toLeftNum -= macroSize;
857 column2PUs[overflowColId].pop_front();
858 }
859 }
860 if (rightAvaliableCapacity - rightUtil > 0)
861 {
862 while (toRightNum > 0 && column2PUs[overflowColId].size() > 0)
863 {
864 column2PUs[overflowColId + 1].push_front(
865 column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
866 int macroSize = getPUSiteNum(column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
867 columnUntilization[overflowColId + 1] += macroSize;
868 columnUntilization[overflowColId] -= macroSize;
869 toRightNum -= macroSize;
870 column2PUs[overflowColId].pop_back();
871 }
872 }
873 }
874 }
875 cell2Column.clear();
876 for (unsigned int colId = 0; colId < column2PUs.size(); colId++)
877 {
878 auto PUs = column2PUs[colId];
879 for (auto curPU : PUs)
880 {
881 cell2Column[curPU] = colId;
882 }
883 }
884}
