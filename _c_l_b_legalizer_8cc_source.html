177 {
178  // final Legalization DP
179  // i th macro (start from 0), j th row (start from 0)
180  // f[i][j] = min(f[i-1][j-row[i]]+HPWLChange[i][j-row[i]+1],f[i][j-1])

192  for (auto &PUDeque : Column2PUs)
193  {

195  {
196  assert(PU2Sites.find(tmpPU) == PU2Sites.end());

202  for (int c = 0; c < colNum; c++)

213  // initialize
214

216  std::vector<std::vector<bool>> fChoice(
217  numPUs, std::vector<bool>(curColSites.size(), 0)); // used for tracing back macro-site mapping
218 
219  int minLastPUTop = -1;
220

222  int heightPURow = getPUSiteNum(curColPU[0]);
223  totalMacroCellNum += heightPURow;

240  }
241  }
242  f[0][j] = minHPWLChange;
243  }
244 
245  // DP to minimize HPWL
246  for (int i = 1; i < numPUs; i++)
247  {
248  float minVal = 100000000.0;
249  int heightPURow = getPUSiteNum(curColPU[i]);
250  totalMacroCellNum += heightPURow;
251  for (unsigned int j = totalMacroCellNum - 1; j < curColSites.size();
252  j++) // j start from heightPURow because PU0 must occupy 1+ site(s)
253  {
254  float curHPWLChange = getHPWLChange(curColPU[i], curColSites[j - heightPURow + 1]);
255  if (curColSites[j]->getSiteY() - curColSites[j - heightPURow + 1]->getSiteY() != heightPURow - 1)
256  {
257  // we need to ensure that there is no occpupied sites in this range
258  curHPWLChange = 1100000000.0;
259  }

261  {
262  fChoice[i][j] = 1;

268  minLastPUTop = j;
269  }
270  }
271  }
272 
273  f[i][j] = std::min(f[i][j - 1], f[i - 1][j - heightPURow] + curHPWLChange);
274  }
275  }

278

281  for (int i = numPUs - 1; i >= 0; i--)

300  }
301  }
302 
303  for (auto &PUDeque : Column2PUs)
304  {
305  for (auto tmpPU : PUDeque)
306  {
307  assert(PU2Sites.find(tmpPU) != PU2Sites.end());
308  auto curSite = PU2Sites[tmpPU][0];

312  PU2Y[tmpPU] = curSite->Y();

317  }

545 {

565  adjList[leftCellId].emplace_back(rightSiteIds[curSite], tmpCost);
566  if (tmpCost < minCost)

578  for (unsigned int leftCellId = 0; leftCellId < PUsToLegalize.size(); leftCellId++)
579  {
580  for (unsigned int tmpId = 0; tmpId < adjList[leftCellId].size(); tmpId++)

707 void CLBLegalizer::updatePUMatchingLocation(bool isRoughLegalization, bool updateDisplacement)
708 {
709  PU2X.clear();
710  PU2Y.clear();
711  PU2Columns.clear();
712  PU2SiteX.clear();
713 
714  PU2LegalSites.clear();
715  MCLBColumn2PUs.clear();
716  if (MCLBColumnNum > 0)
717  {
718  MCLBColumn2PUs.resize(MCLBColumnNum, std::deque<PlacementInfo::PlacementUnit *>(0));
719  }
720  LCLBColumn2PUs.clear();
721  if (LCLBColumnNum > 0)
722  {
723  LCLBColumn2PUs.resize(LCLBColumnNum, std::deque<PlacementInfo::PlacementUnit *>(0));
724  }
725 
726  float tmpAverageDisplacement = 0.0;
727 
728  for (auto matchedPair : PULevelMatching)
729  {
730  auto matchedSite = matchedPair.second;
731  auto curPU = matchedPair.first;
732 
733  if (PU2Y.find(curPU) == PU2Y.end())
734  {
735  PU2SiteX[curPU] = -1;
736  PU2Y[curPU] = 0.0;
737  if (curPU->isMCLB())
738  {
739  MCLBColumn2PUs[matchedSite->getSiteX()].push_back(curPU);
740  }
741  if (curPU->isLCLB())
742  {
743  LCLBColumn2PUs[matchedSite->getSiteX()].push_back(curPU);
744  }
745  }
746 
747  float actualPUX = 0;
748  float actualPUY = 0;
749  // since the matched site is for a single DSP/BRAM cell, we need to find its corresponding placement macro
750  // (with multiple DSPs/BRAMs)
751  actualPUX = matchedSite->X();
752  actualPUY = matchedSite->Y();
753 
754  tmpAverageDisplacement += std::fabs(curPU->X() - matchedSite->X()) + std::fabs(curPU->Y() - matchedSite->Y());
755  assert(isRoughLegalization || PU2SiteX[curPU] == -1 || PU2SiteX[curPU] == matchedSite->getSiteX());
756  PU2SiteX[curPU] = matchedSite->getSiteX();
757  PU2X[curPU] += actualPUX;
758  PU2Y[curPU] += actualPUY;
759  if (PU2Columns.find(curPU) == PU2Columns.end())
760  {
761  PU2Columns[curPU] = std::vector<int>(0);
762  }
763  PU2Columns[curPU].push_back(matchedSite->getSiteX());
764  }
765 
766  tmpAverageDisplacement /= PULevelMatching.size();
767 
768  if (verbose)
769  print_info("CLBLegalizer Macro Cell Average Displacement = " + std::to_string(tmpAverageDisplacement));
770 
771  if (updateDisplacement)
772  {
773  if (isRoughLegalization)
774  roughAverageDisplacement = tmpAverageDisplacement;
775  else
776  fixedColumnAverageDisplacement = tmpAverageDisplacement;
777  }
778 
779  for (auto pairPU2X : PU2X)
780  {
781  int numCells = getPUSiteNum(pairPU2X.first);
782  assert(numCells == 1); // currentlt CLB should only cover one site.
783  PU2X[pairPU2X.first] /= numCells;
784  PU2Y[pairPU2X.first] /= numCells;
785  }
786 
787 #pragma omp parallel for
788  for (int i = 0; i < MCLBColumnNum; i++)
789  sortPUsByPU2Y(MCLBColumn2PUs[i]);
790 
791 #pragma omp parallel for
792  for (int i = 0; i < LCLBColumnNum; i++)
793  sortPUsByPU2Y(LCLBColumn2PUs[i]);
794 
795  placementInfo->setPULegalXY(PU2X, PU2Y);
796 }

798 void CLBLegalizer::spreadPUs(int columnNum, std::vector<int> &columnUntilization,
799  std::vector<std::vector<DeviceInfo::DeviceSite *>> &column2Sites,
800  std::vector<std::deque<PlacementInfo::PlacementUnit *>> &column2PUs,
801  std::map<PlacementInfo::PlacementUnit *, int> &cell2Column)
802 {
803  while (true)
804  {
805  int overflowColId = -1;
806  std::vector<int> accumulationUtil(columnNum, 0), accumulationCapacity(columnNum, 0);
807  accumulationUtil[0] = columnUntilization[0];
808  accumulationCapacity[0] = column2Sites[0].size();
809  for (int colId = 1; colId < columnNum; colId++)
810  {
811  accumulationUtil[colId] = accumulationUtil[colId - 1] + columnUntilization[colId];
812  accumulationCapacity[colId] = accumulationCapacity[colId - 1] + column2Sites[colId].size();
813  }
814 
815  for (int colId = 0; colId < columnNum; colId++)
816  {
817  if (column2Sites[colId].size() < (unsigned int)columnUntilization[colId])
818  {
819  overflowColId = colId;
820  break;
821  }
822  }
823  if (overflowColId < 0)
824  {
825  break;
826  }
827  else
828  {
829  int leftAvaliableCapacity = 0;
830  int rightAvaliableCapacity = 0;
831  int leftUtil = 0;
832  int rightUtil = 0;
833 
834  if (overflowColId > 0)
835  {
836  leftAvaliableCapacity += accumulationCapacity[overflowColId - 1];
837  leftUtil += accumulationUtil[overflowColId - 1];
838  }
839  if (overflowColId < columnNum - 1)
840  {
841  rightAvaliableCapacity += accumulationCapacity[columnNum - 1] - accumulationCapacity[overflowColId];
842  rightUtil += accumulationUtil[columnNum - 1] - accumulationUtil[overflowColId];
843  }
844 
845  int overflowNum = (unsigned int)columnUntilization[overflowColId] - column2Sites[overflowColId].size();
846  int toLeft = 0;
847 
848  int totalAvailableCapacity = rightAvaliableCapacity + leftAvaliableCapacity - leftUtil - rightUtil;
849  assert(totalAvailableCapacity > 0);
850  float toLeftRatio = (float)(leftAvaliableCapacity - leftUtil) / totalAvailableCapacity;
851  int toLeftNum = int((overflowNum * toLeftRatio) + 0.4999);
852  int toRightNum = overflowNum - toLeft;
853 
854  if (leftAvaliableCapacity - leftUtil > 0)
855  {
856  while (toLeftNum > 0 && column2PUs[overflowColId].size() > 0)
857  {
858  column2PUs[overflowColId - 1].push_back(column2PUs[overflowColId][0]);
859  int macroSize = getPUSiteNum(column2PUs[overflowColId][0]);
860  columnUntilization[overflowColId - 1] += macroSize;
861  columnUntilization[overflowColId] -= macroSize;
862  toLeftNum -= macroSize;
863  column2PUs[overflowColId].pop_front();
864  }
865  }
866  if (rightAvaliableCapacity - rightUtil > 0)
867  {
868  while (toRightNum > 0 && column2PUs[overflowColId].size() > 0)
869  {
870  column2PUs[overflowColId + 1].push_front(
871  column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
872  int macroSize = getPUSiteNum(column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
873  columnUntilization[overflowColId + 1] += macroSize;
874  columnUntilization[overflowColId] -= macroSize;
875  toRightNum -= macroSize;
876  column2PUs[overflowColId].pop_back();
877  }
878  }
879  }
880  }
881  cell2Column.clear();
882  for (unsigned int colId = 0; colId < column2PUs.size(); colId++)
883  {
884  auto PUs = column2PUs[colId];
885  for (auto curPU : PUs)
886  {
887  cell2Column[curPU] = colId;
888  }
889  }
890 }
