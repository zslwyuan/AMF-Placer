120 // we have to pack cells in design info into placement units in placement info with packer

131
132 // go through several glable placement iterations to get initial placement
133 globalPlacer = new GlobalPlacer(placementInfo, JSON);
134
135 // enable the timing optimization, start initial placement and global placement.
136 timingOptimizer->enhanceNetWeight_LevelBased(10);
137 globalPlacer->clusterPlacement();
138 globalPlacer->GlobalPlacement_fixedCLB(1, 0.0002);
139 globalPlacer->GlobalPlacement_CLBElements(std::stoi(JSON["GlobalPlacementIteration"]) / 3, false, 5, true,
140 true);
141
142 // create a more fine-grained bin-grid to properly spread the cells
143 placementInfo->createGridBins(2.0, 2.0);
144 placementInfo->adjustLUTFFUtilization(-10, true);
145 globalPlacer->spreading(-1);
146 globalPlacer->GlobalPlacement_CLBElements(std::stoi(JSON["GlobalPlacementIteration"]) * 2 / 9, true, 5, true,
147 true);
148
149 // strengthen the timing optimization, enable area adjustion and relax a bit the pseudo nets for more
150 // flexibility
151 timingOptimizer->clusterLongPathInOneClockRegion(20, 0.5);
152 timingOptimizer->enhanceNetWeight_LevelBased(10);
153 globalPlacer->setNeighborDisplacementUpperbound(3.0);
154 globalPlacer->setPseudoNetWeight(globalPlacer->getPseudoNetWeight() * 0.85);
155
156 print_info("Current Total HPWL = " + std::to_string(placementInfo->updateB2BAndGetTotalHPWL()));
157
158 // pack simple LUT-FF pairs and go through several global placement iterations
159 incrementalBELPacker = new IncrementalBELPacker(designInfo, deviceinfo, placementInfo, JSON);
160 incrementalBELPacker->LUTFFPairing(4.0);
161 incrementalBELPacker->FFPairing(4.0);
162 placementInfo->printStat();
163 print_info("Current Total HPWL = " + std::to_string(placementInfo->updateB2BAndGetTotalHPWL()));
164

166 true, timingOptimizer);

169 globalPlacer->GlobalPlacement_CLBElements(std::stoi(JSON["GlobalPlacementIteration"]) * 2 / 9, true, 5, true,
170 true, timingOptimizer);
171
172 // placementInfo->getDesignInfo()->resetNetEnhanceRatio();
173
174 globalPlacer->GlobalPlacement_CLBElements(std::stoi(JSON["GlobalPlacementIteration"]) / 2, true, 5, true, false,
175 timingOptimizer);
176
177 // currently, some fixed/packed flag cannot be stored in the check-point (TODO)
178 clearSomeAttributesCannotRecord();
179
180 // test the check-point mechanism
181 placementInfo->dumpPlacementUnitInformation("PUInfoBeforeFinalPacking");
182 placementInfo->loadPlacementUnitInformation("PUInfoBeforeFinalPacking.gz");
