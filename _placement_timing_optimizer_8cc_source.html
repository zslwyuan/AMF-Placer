32void PlacementTimingOptimizer::enhanceNetWeight_LevelBased(int levelThr)
33{
34 bool printOut = false;
35 std::string dumpFileName = "optNetInfo.txt";
36 std::ofstream outfile0;
37 if (JSONCfg.find("PlacementTimingOptimizer_EnhanceNetWeightLevelBasedLog") != JSONCfg.end())

41 print_status("PlacementTimingOptimizer: dumping enhanceNetWeight_LevelBased to: " + dumpFileName);
42 outfile0.open(dumpFileName.c_str());
43 assert(outfile0.is_open() && outfile0.good() &&
44 "The path for PlacementTimingOptimizer_EnhanceNetWeightLevelBasedLog does not exist and please check "
45 "your path settings");
46 }
47
48 print_status("PlacementTimingOptimizer: enhanceNetWeight_LevelBased starts.");
49 if (levelThr < 4)
50 return;
51 assert(timingInfo);
52
53 float maxEnhanceRatio = 0;
54 auto timingNodes = timingInfo->getSimplePlacementTimingInfo();
55 for (auto tmpNet : designInfo->getNets())
56 tmpNet->setOverallTimingNetEnhancement(1.0);
57
58 enhanceNetWeight_LevelBased_Cnt++;
59 effectFactor = (enhanceNetWeight_LevelBased_Cnt / 30.0);
60 if (effectFactor < 1)
61 effectFactor = std::pow(effectFactor, 1);
62 else
63 effectFactor = 1;
64
65 for (auto cellA : designInfo->getCells())
66 {
67 if (cellA->isVirtualCell())
68 continue;
69 int targetPathLen = timingNodes[cellA->getCellId()]->getLongestPathLength();
70 if (targetPathLen < levelThr)
71 continue;
72
73 // int pinAIdInNet = 0;
74 for (DesignInfo::DesignPin *curPinA : cellA->getPins())
75 {
76 if (curPinA->getNet())
77 {
78 if (curPinA->isInputPort())
79 continue;
80 // int pinBIdInNet = 0;
81 auto pins = curPinA->getNet()->getPins();
82 int pinNum = pins.size();
83 if (pinNum <= 1 || pinNum >= 1000)
84 continue;
85
86 float enhanceRatio;
87 float overflowRatio = std::pow((float)0.8 * targetPathLen / levelThr, 1);
88 // if (overflowRatio > 10)
89 // overflowRatio = 10;
90 if (pinNum < 200)
91 enhanceRatio = 1.5 * (overflowRatio + 0.005 * pinNum);
92 else
93 enhanceRatio = 1.5 * (overflowRatio + 1);
94
95 enhanceRatio = std::pow(enhanceRatio, effectFactor);
96 if (enhanceRatio > maxEnhanceRatio)
97 maxEnhanceRatio = enhanceRatio;
98 curPinA->getNet()->enhanceOverallTimingNetEnhancement(enhanceRatio);
99
100 if (printOut)
101 {
102 outfile0 << "enhanced net: [ " << curPinA->getName()
103 << "] fanOut:" << curPinA->getNet()->getPins().size() << " by " << enhanceRatio << "\n";
104 }
105 }
106 }
107 }
108 if (printOut)
109 {
110 for (auto cellA : designInfo->getCells())
111 {
112 if (cellA->isVirtualCell())
113 continue;
114 int targetPathLen = timingNodes[cellA->getCellId()]->getLongestPathLength();
115
116 outfile0 << "cell path len: [ " << cellA->getName() << "] = " << targetPathLen << "\n";
117 }
118 outfile0.close();
119 }
120
121 print_status("PlacementTimingOptimizer: enhanceNetWeight_LevelBased done (maxEnhancedRatio=" +
122 std::to_string(maxEnhanceRatio) + ")");
