367float PlacementTimingOptimizer::getSlackThr()
368{
369 assert(placementInfo->getTimingInfo());
370
371 // float maxEnhanceRatio = 0;
372 auto timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo();
373 auto &cellLoc = placementInfo->getCellId2location();
374 assert(cellLoc.size() == timingNodes.size());
375
376 // std::ofstream outfile0("timingOptProc.log");
377
378 // auto deviceInfo = placementInfo->getDeviceInfo();
379
380 std::vector<int> slackCntVec(200, 0);
381 int totalSlackChecked = 0;
382
383 netActualSlackPinNum.clear();
384
385 for (auto curNet : placementInfo->getPlacementNets())
386 {
387 auto designNet = curNet->getDesignNet();
388 if (designNet->checkIsPowerNet() || designNet->checkIsGlobalClock())
389 continue;
390
391 if (curNet->getDriverUnits().size() != 1 || curNet->getUnits().size() <= 1 || curNet->getUnits().size() >= 1000)
392 continue;
393 auto &pins = designNet->getPins();
394 int pinNum = pins.size();
395
396 assert(curNet->getUnits().size() == (unsigned int)pinNum);
397
398 int driverPinInNet = -1;
399
400 for (int i = 0; i < pinNum; i++)
401 {
402 if (pins[i]->isOutputPort())
403 {
404 driverPinInNet = i;
405 break;
406 }
407 }
408
409 assert(driverPinInNet >= 0);
410
411 // get the srcPin information
412 auto srcCell = pins[driverPinInNet]->getCell();
413 unsigned int srcCellId = srcCell->getCellId();
414 auto srcNode = timingNodes[srcCellId];
415 auto srcLoc = cellLoc[srcCellId];
416 // iterate the sinkPin for evaluation and enhancement
417
418 netActualSlackPinNum[curNet] = 0;
419 for (int pinBeDriven = 0; pinBeDriven < pinNum; pinBeDriven++)
420 {
421 if (pinBeDriven == driverPinInNet)
422 continue;
423
424 // get the sinkPin information
425 auto sinkCell = pins[pinBeDriven]->getCell();
426 unsigned int sinkCellId = sinkCell->getCellId();
427 auto sinkNode = timingNodes[sinkCellId];
428 auto sinkLoc = cellLoc[sinkCellId];
429 float netDelay = getDelayByModel(sinkNode, srcNode, sinkLoc.X, sinkLoc.Y, srcLoc.X, srcLoc.Y);
430 float slack = sinkNode->getRequiredArrivalTime() - srcNode->getLatestOutputArrival() - netDelay;
431
432 if (slack > 0)
433 continue;
434
435 netActualSlackPinNum[curNet]++;
436 int slotId = (int)(-slack / 0.1);
437 if (slotId >= 100)
438 slotId = 100;
439 if (slotId < 0)
440 slotId = 0;
441 slackCntVec[slotId] += 1;
442 totalSlackChecked++;
443 }
444 }
445
446 float slackThr = 0;
447 int slackUnderThrCnt = 0;
448 float careRatio = 0.333;
449
450 if (careRatio <= 0.9 && totalSlackChecked > 0)
451 {
452 for (unsigned int i = 0; i < slackCntVec.size(); i++)
453 {
454 slackUnderThrCnt += slackCntVec[i];
455 if (slackUnderThrCnt > totalSlackChecked * (1 - careRatio))
456 break;
457 slackThr -= 0.1;
458 }
459 }
460
461 std::string outputStr = "";
462 for (unsigned int i = 0; i < slackCntVec.size(); i++)
463 {
464 outputStr += " " + std::to_string(slackCntVec[i]);
465 }
466 print_info("Slack distribution:" + outputStr);
467 print_info("slackThr = " + std::to_string(slackThr));
468 return slackThr;

730void PlacementTimingOptimizer::stretchClockRegionColumns()
731{
732 auto &binGrid = placementInfo->getGlobalBinGrid();
733 auto &PU2ClockRegionColumn = placementInfo->getPU2ClockRegionColumn();
734 auto YX2ClockRegion = deviceInfo->getClockRegions();
735 int clockRegionXNum = YX2ClockRegion[0].size();
736 std::vector<int> curClockRegionX2cellNum(clockRegionXNum, 0);
737 std::vector<std::vector<PlacementInfo::PlacementUnit *>> clockRegionX2PUs(
738 clockRegionXNum, std::vector<PlacementInfo::PlacementUnit *>(0));
739 std::vector<float> curClockRegionX2MinY(clockRegionXNum, placementInfo->getGlobalBinMaxLocY());
740 std::vector<float> curClockRegionX2MaxY(clockRegionXNum, placementInfo->getGlobalBinMinLocY());
741
742 for (auto PU : placementInfo->getPlacementUnits())
743 {
744 int clockRegionX, clockRegionY;
745 deviceInfo->getClockRegionByLocation(PU->X(), PU->X(), clockRegionX, clockRegionY);
746 curClockRegionX2cellNum[clockRegionX] += PU->getWeight();
747 clockRegionX2PUs[clockRegionX].push_back(PU);
748
749 int binIdX, binIdY;
750 placementInfo->getGridXY(PU->X(), PU->Y(), binIdX, binIdY);
751
752 if (binGrid[binIdY][binIdX]->getCells().size() > 16)
753 {
754 if (PU->Y() > curClockRegionX2MaxY[clockRegionX])
755 {
756 curClockRegionX2MaxY[clockRegionX] = PU->Y();
757 }
758 if (PU->Y() < curClockRegionX2MinY[clockRegionX])
759 {
760 curClockRegionX2MinY[clockRegionX] = PU->Y();
761 }
762 }
763 }
764
765 std::vector<int> newClockRegionX2cellNum = curClockRegionX2cellNum;
766 for (auto pair : PU2ClockRegionColumn)
767 {
768 auto PU = pair.first;
769 int oriClockRegionX, oriClockRegionY;
770 deviceInfo->getClockRegionByLocation(PU->X(), PU->X(), oriClockRegionX, oriClockRegionY);
771 newClockRegionX2cellNum[oriClockRegionX] -= PU->getWeight();
772 newClockRegionX2cellNum[pair.second] += PU->getWeight();
773 }
774
775 for (unsigned int colX = 0; colX < curClockRegionX2cellNum.size(); colX++)
776 {
777 float stretchRatio = (float)newClockRegionX2cellNum[colX] / (float)curClockRegionX2cellNum[colX];
778
779 float bottomLimit = placementInfo->getGlobalBinMinLocY();
780 float topLimit = placementInfo->getGlobalBinMaxLocY();
781 float completeH = topLimit - bottomLimit;
782 float oriTopY = curClockRegionX2MaxY[colX];
783 float oriBottomY = curClockRegionX2MinY[colX];
784 float oriH = std::abs(oriTopY - oriBottomY);
785 bool maybeCongestion =
786 completeH * 0.625 > oriH && (oriBottomY < 0.075 * completeH || oriTopY > 0.925 * completeH);
787
788 if (oriBottomY < 0.075 * completeH)
789 bottomLimit = std::max(std::max(oriBottomY, bottomLimit), (float)0.05 * completeH);
790 if (oriTopY > 0.925 * completeH)
791 topLimit = std::min(std::min(oriTopY, topLimit), (float)0.95 * completeH);
792
793 if (std::abs(oriTopY - oriBottomY) < 0.1)
794 continue;
795 if (stretchRatio > 1 || (maybeCongestion && stretchRatio > 0.95))
796 {
797 if (maybeCongestion)
798 {
799 print_warning("PlacementTimingOptimizer: " + std::to_string(colX) +
800 "th column is stretched more to avoid potential congestion");
801 stretchRatio *= 1.1;
802 }
803 else
804 stretchRatio *= 1.05;
805 float newH = stretchRatio * oriH;
806 float deltaH = newH - oriH;
807 float newTopY = oriTopY + deltaH / 2;
808 float newBottomY = oriBottomY - deltaH / 2;
809
810 if (newBottomY < bottomLimit)
811 {
812 newTopY += (bottomLimit - newBottomY);
813 newBottomY = bottomLimit;
814 }
815
816 if (newTopY > topLimit)
817 {
818 newBottomY = std::max(bottomLimit, newBottomY - (newTopY - topLimit));
819 newTopY = topLimit;
820 }
821 newH = newTopY - newBottomY;
822 if (oriH < 0.01)
823 continue;
824 stretchRatio = newH / oriH;
825 for (auto PU : clockRegionX2PUs[colX])
826 {
827 if (!PU->isFixed())
828 {
829 float fY = (PU->Y() - oriBottomY) * stretchRatio + newBottomY;
830 float fX = PU->X();
831 placementInfo->legalizeXYInArea(PU, fX, fY);
832 PU->setAnchorLocationAndForgetTheOriginalOne(fX, fY);
833 }
834 }
835 print_warning("PlacementTimingOptimizer: " + std::to_string(colX) + "th column is stretched by " +
836 std::to_string(stretchRatio) + " oriBottomY=" + std::to_string(oriBottomY) +
837 " oriTopY=" + std::to_string(oriTopY) + " newBottomY=" + std::to_string(newBottomY) +
838 " newTopY=" + std::to_string(newTopY));
839 }
840 }
841
842 placementInfo->updateElementBinGrid();
843}

889std::vector<float> &PlacementTimingOptimizer::getPUId2Slack(bool update)
890{
891 if (update)
892 {
893 auto timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo();
894 // float clockPeriod = placementInfo->getTimingInfo()->getSimplePlacementTimingGraph()->getClockPeriod();
895 auto &cellLoc = placementInfo->getCellId2location();
896 assert(cellLoc.size() == timingNodes.size());
897
898 unsigned int highFanoutThr = 1000;
899
900 int PUNum = placementInfo->getPlacementUnits().size();
901 PUId2Slack.clear();
902 PUId2Slack.resize(PUNum, 0);
903
904#pragma omp parallel for
905 for (int PUId = 0; PUId < PUNum; PUId++)
906 {
907 auto &curPU = placementInfo->getPlacementUnits()[PUId];
908
909 for (auto curNet : *(curPU->getNetsSetPtr()))
910 {
911 if (curNet->getDriverUnits().size() == 0)
912 continue;
913 if (curNet->getDriverUnits()[0] != curPU)
914 continue;
915
916 auto designNet = curNet->getDesignNet();
917 assert(designNet);
918 if (designNet->checkIsPowerNet() || designNet->checkIsGlobalClock())
919 continue;
920
921 if (curNet->getDriverUnits().size() != 1 || curNet->getUnits().size() <= 1 ||
922 curNet->getUnits().size() >= highFanoutThr)
923 continue;
924 auto &PUs = curNet->getUnits();
925 auto &pins = designNet->getPins();
926 int pinNum = pins.size();
927
928 assert(curNet->getUnits().size() == (unsigned int)pinNum);
929
930 int driverPinInNet = -1;
931
932 for (int i = 0; i < pinNum; i++)
933 {
934 if (pins[i]->isOutputPort())
935 {
936 driverPinInNet = i;
937 break;
938 }
939 }
940
941 assert(driverPinInNet >= 0);
942
943 // get the srcPin information
944 auto srcCell = pins[driverPinInNet]->getCell();
945 unsigned int srcCellId = srcCell->getCellId();
946 auto srcNode = timingNodes[srcCellId];
947 auto srcLoc = cellLoc[srcCellId];
948
949 for (int pinBeDriven = 0; pinBeDriven < pinNum; pinBeDriven++)
950 {
951 if (pinBeDriven == driverPinInNet)
952 continue;
953
954 // get the sinkPin information
955 auto sinkCell = pins[pinBeDriven]->getCell();
956 unsigned int sinkCellId = sinkCell->getCellId();
957 auto sinkNode = timingNodes[sinkCellId];
958 auto sinkLoc = cellLoc[sinkCellId];
959
960 float netDelay = getDelayByModel(sinkNode, srcNode, sinkLoc.X, sinkLoc.Y, srcLoc.X, srcLoc.Y);
961 float slack = sinkNode->getRequiredArrivalTime() - srcNode->getLatestOutputArrival() - netDelay;
962 if (slack < 0)
963 {
964 if (slack < PUId2Slack[PUId])
965 PUId2Slack[PUId] = slack;
966 }
967 }
968 }
969 }
970 }
971 return PUId2Slack;
