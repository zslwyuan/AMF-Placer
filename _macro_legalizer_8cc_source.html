948void MacroLegalizer::spreadMacros(int columnNum, std::vector<int> &columnUntilization,
949 std::vector<std::vector<DeviceInfo::DeviceSite *>> &column2Sites,
950 std::vector<std::deque<PlacementInfo::PlacementUnit *>> &column2PUs,
951 std::map<DesignInfo::DesignCell *, int> &cell2Column, float budgeRatio)
952{
953 while (true)
954 {
955 int overflowColId = -1;
956 std::vector<int> accumulationUtil(columnNum, 0), accumulationCapacity(columnNum, 0);
957 accumulationUtil[0] = columnUntilization[0];
958 accumulationCapacity[0] = column2Sites[0].size() * budgeRatio;
959 for (int colId = 1; colId < columnNum; colId++)
960 {
961 accumulationUtil[colId] = accumulationUtil[colId - 1] + columnUntilization[colId];
962 accumulationCapacity[colId] = accumulationCapacity[colId - 1] + column2Sites[colId].size() * budgeRatio;
963 }
964
965 for (int colId = 0; colId < columnNum; colId++)
966 {
967 if (budgeRatio == 1)
968 {
969 if (column2Sites[colId].size() < (unsigned int)columnUntilization[colId])
970 {
971 overflowColId = colId;
972 break;
973 }
974 }
975 else
976 {
977 if (column2Sites[colId].size() * (budgeRatio + 0.05) < (unsigned int)columnUntilization[colId])
978 {
979 overflowColId = colId;
980 break;
981 }
982 }
983 }
984 if (overflowColId < 0)
985 {
986 break;
987 }
988 else
989 {
990 print_warning("column overflow resolving");
991 for (int colId = 0; colId < columnNum; colId++)
992 {
993 std::cout << " colId#" << colId << " columnUntilization:" << columnUntilization[colId] << " "
994 << " siteCap:" << column2Sites[colId].size() << "\n";
995 }
996 int leftAvaliableCapacity = 0;
997 int rightAvaliableCapacity = 0;
998 int leftUtil = 0;
999 int rightUtil = 0;
1000
1001 if (overflowColId > 0)
1002 {
1003 leftAvaliableCapacity += accumulationCapacity[overflowColId - 1];
1004 leftUtil += accumulationUtil[overflowColId - 1];
1005 }
1006 if (overflowColId < columnNum - 1)
1007 {
1008 rightAvaliableCapacity += accumulationCapacity[columnNum - 1] - accumulationCapacity[overflowColId];
1009 rightUtil += accumulationUtil[columnNum - 1] - accumulationUtil[overflowColId];
1010 }
1011
1012 int overflowNum = (unsigned int)columnUntilization[overflowColId] -
1013 column2Sites[overflowColId].size() * budgeRatio; // spread more for redundant space
1014 int toLeft = 0;
1015
1016 int totalAvailableCapacity = rightAvaliableCapacity + leftAvaliableCapacity - leftUtil - rightUtil;
1017 assert(totalAvailableCapacity > 0);
1018 float toLeftRatio = (float)(leftAvaliableCapacity - leftUtil) / totalAvailableCapacity;
1019 int toLeftNum = int((overflowNum * toLeftRatio) + 0.4999);
1020 int toRightNum = overflowNum - toLeft;
1021
1022 if (leftAvaliableCapacity - leftUtil > 0)
1023 {
1024 while (toLeftNum > 0 && column2PUs[overflowColId].size() > 0)
1025 {
1026 column2PUs[overflowColId - 1].push_back(column2PUs[overflowColId][0]);
1027 int macroSize = getMarcroCellNum(column2PUs[overflowColId][0]);
1028 columnUntilization[overflowColId - 1] += macroSize;
1029 columnUntilization[overflowColId] -= macroSize;
1030 toLeftNum -= macroSize;
1031 column2PUs[overflowColId].pop_front();
1032 }
1033 }
1034 if (rightAvaliableCapacity - rightUtil > 0)
1035 {
1036 while (toRightNum > 0 && column2PUs[overflowColId].size() > 0)
1037 {
1038 column2PUs[overflowColId + 1].push_front(
1039 column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
1040 int macroSize = getMarcroCellNum(column2PUs[overflowColId][column2PUs[overflowColId].size() - 1]);
1041 columnUntilization[overflowColId + 1] += macroSize;
1042 columnUntilization[overflowColId] -= macroSize;
1043 toRightNum -= macroSize;
1044 column2PUs[overflowColId].pop_back();
1045 }
1046 }
1047 }
1048 }
1049 cell2Column.clear();
1050 for (unsigned int colId = 0; colId < column2PUs.size(); colId++)
1051 {
1052 auto PUs = column2PUs[colId];
1053 for (auto curPU : PUs)
1054 {
1055 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(curPU))
1056 {
1057 auto curCell = unpackedCell->getCell();
1058 cell2Column[curCell] = colId;
1059 }
1060 else if (auto macroPU = dynamic_cast<PlacementInfo::PlacementMacro *>(curPU))
1061 {
1062 for (auto curCell : macroPU->getCells())
1063 {
1064 cell2Column[curCell] = colId;
1065 }
1066 }
1067 }
1068 }
