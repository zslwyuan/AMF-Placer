684 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
685 {
686 float offsetXInRegion = x - leftX;
687 float offsetYInRegion = y - bottomY;
688
689 // this approach is not accurate enought since the slice columns are not evenly
690 // distributed in a clock region. However, it is fast for frequently checking of
691 // clock utilization
692 int colX = offsetXInRegion / colWidth;
693 int colY = offsetYInRegion / colHeight;
694 assert(colX >= -1);
695 assert(colY >= -1);
696 assert(clockColumns.size());
697 assert(colY <= (int)clockColumns.size());
698 assert(colX <= (int)clockColumns[0].size());
699 if (colX < 0)
700 colX = 0;
701 if (colY < 0)
702 colY = 0;
703 if (colX >= (int)clockColumns[0].size())
704 {
705 colX = clockColumns[0].size() - 1;
706 }
707 if (colY >= (int)clockColumns.size())
708 {
709 colY = clockColumns.size() - 1;
710 }
711 clockColumns[colY][colX]->addClockNetId(clockNetId, cellId);
712 }
