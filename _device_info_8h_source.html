790 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
791 {
792 float offsetXInRegion = x - leftX;
793 float offsetYInRegion = y - bottomY;
794
795 // this approach is not accurate enought since the slice columns are not evenly
796 // distributed in a clock region. However, it is fast for frequently checking of
797 // clock utilization
798 int colX = offsetXInRegion / colWidth;
799 int colY = offsetYInRegion / colHeight;
800 assert(colX >= -1);
801 assert(colY >= -1);
802 assert(clockColumns.size());
803 assert(colY <= (int)clockColumns.size());
804 assert(colX <= (int)clockColumns[0].size());
805 if (colX < 0)
806 colX = 0;
807 if (colY < 0)
808 colY = 0;
809 if (colX >= (int)clockColumns[0].size())
810 {
811 colX = clockColumns[0].size() - 1;
812 }
813 if (colY >= (int)clockColumns.size())
814 {
815 colY = clockColumns.size() - 1;
816 }
817 clockColumns[colY][colX]->addClockNetId(clockNetId, cellId);
818 }
