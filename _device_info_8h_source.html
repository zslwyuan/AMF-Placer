700 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
701 {
702 float offsetXInRegion = x - leftX;
703 float offsetYInRegion = y - bottomY;
704
705 // this approach is not accurate enought since the slice columns are not evenly
706 // distributed in a clock region. However, it is fast for frequently checking of
707 // clock utilization
708 int colX = offsetXInRegion / colWidth;
709 int colY = offsetYInRegion / colHeight;
710 assert(colX >= -1);
711 assert(colY >= -1);
712 assert(clockColumns.size());
713 assert(colY <= (int)clockColumns.size());
714 assert(colX <= (int)clockColumns[0].size());
715 if (colX < 0)
716 colX = 0;
717 if (colY < 0)
718 colY = 0;
719 if (colX >= (int)clockColumns[0].size())
720 {
721 colX = clockColumns[0].size() - 1;
722 }
723 if (colY >= (int)clockColumns.size())
724 {
725 colY = clockColumns.size() - 1;
726 }
727 clockColumns[colY][colX]->addClockNetId(clockNetId, cellId);
728 }
