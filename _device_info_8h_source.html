668 inline void addClockAndCell(int clockNetId, int cellId, float x, float y)
669 {
670 float offsetXInRegion = x - leftX;
671 float offsetYInRegion = y - bottomY;
672
673 // this approach is not accurate enought since the slice columns are not evenly
674 // distributed in a clock region. However, it is fast for frequently checking of
675 // clock utilization
676 int colX = offsetXInRegion / colWidth;
677 int colY = offsetYInRegion / colHeight;
678 assert(colX >= -1);
679 assert(colY >= -1);
680 assert(clockColumns.size());
681 assert(colY <= (int)clockColumns.size());
682 assert(colX <= (int)clockColumns[0].size());
683 if (colX < 0)
684 colX = 0;
685 if (colY < 0)
686 colY = 0;
687 if (colX >= (int)clockColumns[0].size())
688 {
689 colX = clockColumns[0].size() - 1;
690 }
691 if (colY >= (int)clockColumns.size())
692 {
693 colY = clockColumns.size() - 1;
694 }
695 clockColumns[colY][colX].addClockNetId(clockNetId, cellId);
696 }
