581 void findPU2SitesInDistance()
582 {
583 PU2SitesInDisplacementThreshold.clear();
584
585 int PUsNum = PUsToLegalize.size();
586
587 for (int i = 0; i < PUsNum; i++)
588 {
589 PU2SitesInDisplacementThreshold[PUsToLegalize[i]] = nullptr;
590 }
591
592#pragma omp parallel for
593 for (int i = 0; i < PUsNum; i++)
594 {
595 PlacementInfo::PlacementUnit *curPU = PUsToLegalize[i];
596 DesignInfo::DesignCell *curCell = nullptr;
597 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(curPU))
598 {
599 curCell = unpackedCell->getCell();
600 }
601 else if (auto curMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(curPU))
602 {
603 assert(curMacro->getCells().size() > 0);
604 curCell = curMacro->getCells()[0];
605 }
606 assert(curCell);
607 PU2SitesInDisplacementThreshold[curPU] = placementInfo->findNeiborSiteFromBinGrid(
608 curCell, cellLoc[curCell->getCellId()].X, cellLoc[curCell->getCellId()].Y, displacementThreshold,
609 candidateFactor * maxNumCandidate);
610 }

663 inline float getHPWLChange(PlacementInfo::PlacementUnit *curPU, DeviceInfo::DeviceSite *curSite)
664 {
665 assert(PU2Site2HPWLIncrease.find(curPU) != PU2Site2HPWLIncrease.end());
666 if (PU2Site2HPWLIncrease[curPU].find(curSite) != PU2Site2HPWLIncrease[curPU].end())
667 {
668 return PU2Site2HPWLIncrease[curPU][curSite];
669 }
670 float oriHPWL = 0.0;
671 float newHPWL = 0.0;
672 float PUX = curSite->X();
673 float PUY = curSite->Y();
674 auto nets = placementInfo->getPlacementUnitId2Nets()[curPU->getId()];
675
676 for (auto curNet : nets)
677 {
678 if (curNet->getDesignNet()->getPins().size() > 1000) // it could be clock
679 continue;
680 oriHPWL += curNet->getHPWL(y2xRatio);
681 newHPWL += curNet->getNewHPWLByTrying(curPU, PUX, PUY, y2xRatio);
682 }
683
684 PU2Site2HPWLIncrease[curPU][curSite] = (newHPWL - oriHPWL);
685 return (newHPWL - oriHPWL);
686 }

721 inline int sortPartition(PlacementInfo::PlacementUnit *curPU, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
722 int high)
723 {
724 int pivot, index, i;
725 index = low;
726 pivot = high;
727 for (i = low; i < high; i++)
728 {
729 // finding index of pivot.
730 // if (a[i] < a[pivot])
731 // if (getDisplacement(PULoc, sites[i]) < getDisplacement(PULoc, sites[pivot]))
732 if (getHPWLChange(curPU, sites[i]) < getHPWLChange(curPU, sites[pivot]))
733 {
734 swapSitePtr(&sites[i], &sites[index]);
735 index++;
736 }
737 }
738 swapSitePtr(&sites[pivot], &sites[index]);
739 return index;
