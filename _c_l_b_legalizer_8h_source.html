650 inline float getHPWLChange(PlacementInfo::PlacementUnit *curPU, DeviceInfo::DeviceSite *curSite)
651 {
652 assert(PU2Site2HPWLIncrease.find(curPU) != PU2Site2HPWLIncrease.end());
653 if (PU2Site2HPWLIncrease[curPU].find(curSite) != PU2Site2HPWLIncrease[curPU].end())
654 {
655 return PU2Site2HPWLIncrease[curPU][curSite];
656 }
657 float oriHPWL = 0.0;
658 float newHPWL = 0.0;
659 float PUX = curSite->X();
660 float PUY = curSite->Y();
661 auto nets = placementInfo->getPlacementUnitId2Nets()[curPU->getId()];
662
663 for (auto curNet : nets)
664 {
665 if (curNet->getDesignNet()->getPins().size() > 1000) // it could be clock
666 continue;
667 oriHPWL += curNet->getHPWL(y2xRatio);
668 newHPWL += curNet->getNewHPWLByTrying(curPU, PUX, PUY, y2xRatio);
669 }
670
671 PU2Site2HPWLIncrease[curPU][curSite] = (newHPWL - oriHPWL);
672 return (newHPWL - oriHPWL);
673 }

708 inline int sortPartition(PlacementInfo::PlacementUnit *curPU, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
709 int high)
710 {
711 int pivot, index, i;
712 index = low;
713 pivot = high;
714 for (i = low; i < high; i++)
715 {
716 // finding index of pivot.
717 // if (a[i] < a[pivot])

719 if (getHPWLChange(curPU, sites[i]) < getHPWLChange(curPU, sites[pivot]))
