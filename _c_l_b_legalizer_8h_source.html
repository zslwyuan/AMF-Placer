588 void findPU2SitesInDistance()
589 {
590 PU2SitesInDisplacementThreshold.clear();
591
592 int PUsNum = PUsToLegalize.size();
593
594 for (int i = 0; i < PUsNum; i++)
595 {
596 PU2SitesInDisplacementThreshold[PUsToLegalize[i]] = nullptr;
597 }
598
599#pragma omp parallel for
600 for (int i = 0; i < PUsNum; i++)
601 {
602 PlacementInfo::PlacementUnit *curPU = PUsToLegalize[i];
603 DesignInfo::DesignCell *curCell = nullptr;
604 if (auto unpackedCell = dynamic_cast<PlacementInfo::PlacementUnpackedCell *>(curPU))
605 {
606 curCell = unpackedCell->getCell();
607 }
608 else if (auto curMacro = dynamic_cast<PlacementInfo::PlacementMacro *>(curPU))
609 {
610 assert(curMacro->getCells().size() > 0);
611 curCell = curMacro->getCells()[0];
612 }
613 assert(curCell);
614 PU2SitesInDisplacementThreshold[curPU] = placementInfo->findNeiborSiteFromBinGrid(
615 curCell, cellLoc[curCell->getCellId()].X, cellLoc[curCell->getCellId()].Y, displacementThreshold,
616 candidateFactor * maxNumCandidate);
617 }

670 inline float getHPWLChange(PlacementInfo::PlacementUnit *curPU, DeviceInfo::DeviceSite *curSite)
671 {
672 assert(PU2Site2HPWLIncrease.find(curPU) != PU2Site2HPWLIncrease.end());
673 if (PU2Site2HPWLIncrease[curPU].find(curSite) != PU2Site2HPWLIncrease[curPU].end())
674 {
675 return PU2Site2HPWLIncrease[curPU][curSite];
676 }
677 float oriHPWL = 0.0;
678 float newHPWL = 0.0;
679 float PUX = curSite->X();
680 float PUY = curSite->Y();
681 auto nets = placementInfo->getPlacementUnitId2Nets()[curPU->getId()];
682
683 for (auto curNet : nets)
684 {
685 if (curNet->getDesignNet()->getPins().size() > 10000) // it could be clock
686 continue;
687 oriHPWL += curNet->getHPWL(y2xRatio);
688 newHPWL += curNet->getNewHPWLByTrying(curPU, PUX, PUY, y2xRatio);
689 }
690
691 PU2Site2HPWLIncrease[curPU][curSite] = (newHPWL - oriHPWL);
692 return (newHPWL - oriHPWL);
693 }

728 inline int sortPartition(PlacementInfo::PlacementUnit *curPU, std::vector<DeviceInfo::DeviceSite *> &sites, int low,
729 int high)
730 {
731 int pivot, index, i;
732 index = low;
733 pivot = high;
734 for (i = low; i < high; i++)
735 {
736 // finding index of pivot.
737 // if (a[i] < a[pivot])
738 // if (getDisplacement(PULoc, sites[i]) < getDisplacement(PULoc, sites[pivot]))
739 if (getHPWLChange(curPU, sites[i]) < getHPWLChange(curPU, sites[pivot]))
740 {
741 swapSitePtr(&sites[i], &sites[index]);
742 index++;
743 }
744 }
745 swapSitePtr(&sites[pivot], &sites[index]);
746 return index;
