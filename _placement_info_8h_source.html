2828 bool findLUT = curCell->isLUT();

2830 float targetY = cellId2location[curCell->getCellId()].Y;
2831 std::vector<DesignInfo::DesignCell *> *res = new std::vector<DesignInfo::DesignCell *>();
2832 res->clear();
2833
2834 int binIdX, binIdY;
2835 getGridXY(targetX, targetY, binIdX, binIdY);
2836
2837 assert(binIdY >= 0);
2838 assert((unsigned int)binIdY < LUTFFBinGrid.size());
2839 assert(binIdX >= 0);
2840 assert((unsigned int)binIdX < LUTFFBinGrid[binIdY].size());
2841 assert(LUTFFBinGrid[binIdY][binIdX]->inRange(targetX, targetY));
2842
2843 std::queue<std::pair<int, int>> binXYqueue;
2844 std::set<std::pair<int, int>> reachedBinXYs;
2845 binXYqueue.emplace(binIdX, binIdY);
2846 reachedBinXYs.emplace(binIdX, binIdY);
2847
2848 bool findItself = false;
2849
2850 while (binXYqueue.size() > 0)
2851 {
2852 std::pair<int, int> curXY = binXYqueue.front();
2853 binXYqueue.pop();
2854 int curbinIdX = curXY.first, curbinIdY = curXY.second;
2855

2857 float bin2TargetXYDistance = curBin->getManhattanDistanceTo(targetX, targetY);
2858 if (bin2TargetXYDistance > displacementUpperbound)
2859 continue;
2860 for (auto tmpCell : curBin->getCells())
2861 {
2862 if ((tmpCell->isLUT() && findLUT) || (tmpCell->isFF() && !findLUT))

2867 if (tmpPUDis <= displacementUpperbound)

2870 {
2871 findItself = true;
2872 }
2873 res->push_back(tmpCell);
2874 }
2875 }
2876 }
2877 assert(findItself);
2878 for (int nextY = curbinIdY - 1; nextY <= curbinIdY + 1; nextY++)
2879 {
2880 for (int nextX = curbinIdX - 1; nextX <= curbinIdX + 1; nextX++)

2885 continue;
