422void WirelengthOptimizer::addPseudoNet_SlackBased(float timingWeight, double slackPowFactor,
423 PlacementTimingOptimizer *timingOptimizer, bool calculate)
424{
425 if (calculate)
426 {
427 for (auto list : PNetId2SlackEnhanceTuples)
428 {
429 if (list)
430 delete list;
431 }
432 PNetId2SlackEnhanceTuples.clear();
433 PNetId2SlackEnhanceTuples.resize(placementInfo->getPlacementNets().size(), nullptr);
434
435 assert(placementInfo->getTimingInfo());
436 if (slackPowFactor < 0 || timingWeight < 0)
437 return;
438
439 // float maxEnhanceRatio = 0;
440 auto timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo();
441 // float clockPeriod = placementInfo->getTimingInfo()->getSimplePlacementTimingGraph()->getClockPeriod();
442 auto &cellLoc = placementInfo->getCellId2location();
443 assert(cellLoc.size() == timingNodes.size());
444 auto &netActualSlackPinNum = timingOptimizer->getNetActualSlackPinNum();
445
446 int enhanceNetCnt = 0;
447 unsigned int highFanoutThr = 10000;
448
449 if (placementInfo->getNetDistributionByDensity(512) < 200)
450 {
451 highFanoutThr = 1000;
452 print_warning("highFanoutThr is set to 1000");
453 }
454
455 // int targetCellId = placementInfo->getDesignInfo()->getCell(targetCellName)->getCellId();
456
457 int netNum = placementInfo->getPlacementNets().size();
458
459#pragma omp parallel for
460 for (int PNetId = 0; PNetId < netNum; PNetId++)
461 {
462 PNetId2SlackEnhanceTuples[PNetId] = new std::vector<slackEnhanceTuple>();
463 auto curNet = placementInfo->getPlacementNets()[PNetId];
464 assert(curNet);
465 auto designNet = curNet->getDesignNet();
466 assert(designNet);
467 assert(curNet->getId() < PNetId2SlackEnhanceTuples.size());
468 assert(curNet->getId() == PNetId);
469 if (designNet->checkIsPowerNet() || designNet->checkIsGlobalClock())
470 continue;
471
472 if (curNet->getDriverUnits().size() != 1 || curNet->getUnits().size() <= 1 ||
473 curNet->getUnits().size() >= highFanoutThr)
474 continue;
475 auto &PUs = curNet->getUnits();
476 auto &pins = designNet->getPins();
477 int pinNum = pins.size();
478
479 assert(curNet->getUnits().size() == (unsigned int)pinNum);
480
481 int driverPinInNet = -1;
482
483 for (int i = 0; i < pinNum; i++)
484 {
485 if (pins[i]->isOutputPort())
486 {
487 driverPinInNet = i;
488 break;
489 }
490 }
491
492 assert(driverPinInNet >= 0);
493
494 // get the srcPin information
495 auto srcCell = pins[driverPinInNet]->getCell();
496 unsigned int srcCellId = srcCell->getCellId();
497 auto srcNode = timingNodes[srcCellId];
498 auto srcLoc = cellLoc[srcCellId];
499 int driverPathLen = timingNodes[srcCellId]->getLongestPathLength();
500
501 if (netActualSlackPinNum[curNet] == 0)
502 continue;
503
504 float w = 2 * timingWeight / std::pow((float)(netActualSlackPinNum[curNet]), 0.5);
505
506 if (srcCell->getCellId() == targetCellId)
507 {
508 std::cout << "driver: " << targetCellName << " x: " << srcLoc.X << " y: " << srcLoc.Y << "\n";
509 }
510 auto &pinEnhanceRate = netPinEnhanceRate[designNet];
511 // iterate the sinkPin for evaluation and enhancement
512 float timingEffect = timingOptimizer->getEffectFactor();
513 for (int pinBeDriven = 0; pinBeDriven < pinNum; pinBeDriven++)
514 {
515 if (pinBeDriven == driverPinInNet)
516 continue;
517
518 // get the sinkPin information
519 auto sinkCell = pins[pinBeDriven]->getCell();
520 unsigned int sinkCellId = sinkCell->getCellId();
521 auto sinkNode = timingNodes[sinkCellId];
522 auto sinkLoc = cellLoc[sinkCellId];
523 int succPathLen = sinkNode->getLongestPathLength();
524 float netDelay =
525 timingOptimizer->getDelayByModel(sinkNode, srcNode, sinkLoc.X, sinkLoc.Y, srcLoc.X, srcLoc.Y);
526 float slack = sinkNode->getRequiredArrivalTime() - srcNode->getLatestOutputArrival() - netDelay;
527 float clockPeriod = sinkNode->getClockPeriod();
528 if (clockPeriod < 0)
529 {
530 clockPeriod = placementInfo->getTimingInfo()->getSimplePlacementTimingGraph()->getClockPeriod();
531 std::cout << sinkCell << " has no clock setting!!\n";
532 }
533 if (slack > 0)
534 continue;
535 enhanceNetCnt++;
536 // enhance the net based on the slack
537 float enhanceRatio = std::pow(1 - slack / clockPeriod, slackPowFactor);
538 if (slack < slackThr)
539 {
540 enhanceRatio = std::pow(enhanceRatio, slack / slackThr * 3);
541 }
542 // * std::pow(netDelay / expectedAvgDelay_driver, 0.6);
543
544 if (timingOptimizer->getEffectFactor() < 0.5)
545 {
546 if (srcNode->checkIsRegister())
547 {
548 if (succPathLen > 0)
549 {
550 float expectedAvgDelay_succ = clockPeriod / succPathLen;
551 if (netDelay > expectedAvgDelay_succ)
552 enhanceRatio =
553 std::max(enhanceRatio, (float)std::pow((netDelay / expectedAvgDelay_succ), 0.66));
554 }
555 }
556 else
557 {
558 if (driverPathLen > 0)
559 {
560 float expectedAvgDelay_driver = clockPeriod / driverPathLen;
561 if (netDelay > expectedAvgDelay_driver)
562 enhanceRatio =
563 std::max(enhanceRatio, (float)std::pow((netDelay / expectedAvgDelay_driver), 0.66));
564 }
565 }
566 }
567
568 if (pinEnhanceRate[pinBeDriven] < 0)
569 {
570 pinEnhanceRate[pinBeDriven] = enhanceRatio;
571 }
572 else
573 {
574 if (enhanceRatio > 0)
575 {
576 float shrinkRatio = pinEnhanceRate[pinBeDriven] / enhanceRatio;
577 if (timingOptimizer->getEffectFactor() >= 1)
578 {
579 if (shrinkRatio > 300)
580 {
581 enhanceRatio = std::pow(enhanceRatio, 0.1) * std::pow(pinEnhanceRate[pinBeDriven], 0.9);
582 }
583 else if (shrinkRatio > 200)
584 {
585 enhanceRatio = std::pow(enhanceRatio, 0.2) * std::pow(pinEnhanceRate[pinBeDriven], 0.8);
586 }
587 else if (shrinkRatio > 100)
588 {
589 enhanceRatio =
590 std::pow(enhanceRatio, 0.33) * std::pow(pinEnhanceRate[pinBeDriven], 0.67);
591 }
592 else if (shrinkRatio > 64)
593 {
594 enhanceRatio = std::pow(enhanceRatio, 0.4) * std::pow(pinEnhanceRate[pinBeDriven], 0.6);
595 }
596 }
597 pinEnhanceRate[pinBeDriven] = enhanceRatio;
598 }
599 }
600 if (srcCell->getCellId() == targetCellId)
601 {
602 std::cout << "sink: " << sinkCell->getName() << " x: " << sinkLoc.X << " y: " << sinkLoc.Y
603 << " netDelay: " << netDelay << " slack: " << slack << " w: " << w
604 << " enhanceRatio: " << enhanceRatio << "\n";
605 }
606 if (sinkCell->getCellId() == targetCellId)
607 {
