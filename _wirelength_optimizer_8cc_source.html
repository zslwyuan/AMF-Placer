170void WirelengthOptimizer::updateB2BNetWeight(float pesudoNetWeight, bool enableMacroPseudoNet2Site, bool considerNetNum,
171 bool enableUserDefinedClusterOpt,
172 PlacementTimingOptimizer *timingOptimizer)
173{
174 if (verbose)
175 print_status("update B2B Net Weight Start.");
176
177 bool updateTrue = true, updateFalse = false;
178 std::thread t1(updateB2BNetWeightWorker, std::ref(placementInfo),
179 std::ref(xSolver->solverData.objectiveMatrixTripletList),
180 std::ref(xSolver->solverData.objectiveMatrixDiag), std::ref(xSolver->solverData.objectiveVector),
181 std::ref(generalNetWeight), std::ref(y2xRatio), std::ref(updateTrue), std::ref(updateFalse));
182 std::thread t2(updateB2BNetWeightWorker, std::ref(placementInfo),
183 std::ref(ySolver->solverData.objectiveMatrixTripletList),
184 std::ref(ySolver->solverData.objectiveMatrixDiag), std::ref(ySolver->solverData.objectiveVector),
185 std::ref(generalNetWeight), std::ref(y2xRatio), std::ref(updateFalse), std::ref(updateTrue));
186 t1.join();
187 t2.join();
188
189 if (enableMacroPseudoNet2Site && !directMacroLegalize)
190 {
191 addPseudoNetForMacros(pesudoNetWeight, considerNetNum);
192 }
193 else
194 {
195 print_warning("addPseudoNetForMacros is disabled according to placer configuration.");
196 }
197
198 if (timingOptimizer)
199 {
200 addPseudoNet_LevelBased(placementInfo->getLongPathThresholdLevel(),
201 (0.2 * timingOptimizer->getEffectFactor()) * generalNetWeight, 25);
202 }
203
204 if (enableUserDefinedClusterOpt)
205 {
206 updatePseudoNetForUserDefinedClusters(pesudoNetWeight);
207 }
208
209 addPseudoNet2LoctionForAllPUs(pesudoNetWeight, considerNetNum);
210 updatePseudoNetForClockRegion(0.2 * pesudoNetWeight);
211
212 if (verbose)
213 print_status("update B2B Net Weight Done.");
214
215 if (JSONCfg.find("DrawNetAfterEachIteration") != JSONCfg.end())
216 {
217 if (JSONCfg["DrawNetAfterEachIteration"] == "true")
218 {
219 int netId;
220 std::cout << "please input the ID of net you want to show (-1 for stopping):\n";
221 std::cin >> netId;
222 while (netId >= 0)
223 {
224 auto net = placementInfo->getPlacementNets()[netId];
225 net->drawNet();
226 std::cout << "please input the ID of net you want to show (-1 for stopping):\n";
227 std::cin >> netId;
228 }
229 }
230 }

254void WirelengthOptimizer::addPseudoNetForMacros(float pesudoNetWeight, bool considerNetNum)
255{
256 std::map<PlacementInfo::PlacementUnit *, float> &PUX = placementInfo->getPULegalXY().first;
257 std::map<PlacementInfo::PlacementUnit *, float> &PUY = placementInfo->getPULegalXY().second;
258 macroPseudoNetCnt++;
259
260 if (placementInfo->getProgress() > 0.8)
261 oriMacroLegalizationWeight *= 1.025;
262 if (placementInfo->getProgress() > 0.90)
263 oriMacroLegalizationWeight *= 1.05;
264 if (macroPseudoNetCnt > 40)
265 oriMacroLegalizationWeight *= 1.05;

267 print_info("WLOptimizer macroLegalizePseudoNet=" + std::to_string(macroPseudoNetFactor * pesudoNetWeight));
268 if (considerNetNum)
269 {
270 for (auto pairPUX : PUX)
271 {

289 false); // DSP-BRAM-like element
290 }
291 for (auto pairPUY : PUY)

332void WirelengthOptimizer::addPseudoNet_LevelBased(int levelThr, float timingWeight, double disExpected)
333{
334 assert(placementInfo->getTimingInfo());
335 if (levelThr < 4)
336 return;
337
338 float maxEnhanceRatio = 0;
339 auto timingNodes = placementInfo->getTimingInfo()->getSimplePlacementTimingInfo();
340 auto &cellLoc = placementInfo->getCellId2location();
341 assert(cellLoc.size() == timingNodes.size());
342
343 float powFactor = 0.5 + 0.2 * placementInfo->getProgress();
344 // std::ofstream outfile0("timingOptProc.log");
345
346 auto deviceInfo = placementInfo->getDeviceInfo();
347
348 for (auto curNet : placementInfo->getPlacementNets())
349 {
350 auto designNet = curNet->getDesignNet();
351 if (designNet->checkIsPowerNet())
352 continue;
353
354 if (curNet->getDriverUnits().size() != 1 || curNet->getUnits().size() <= 1 || curNet->getUnits().size() >= 500)
355 continue;
356 auto &PUs = curNet->getUnits();
357 auto &pins = designNet->getPins();
358 int pinNum = pins.size();
359
360 assert(curNet->getUnits().size() == (unsigned int)pinNum);
361
362 int driverPinInNet = -1;
363
364 for (int i = 0; i < pinNum; i++)
365 {
366 if (pins[i]->isOutputPort())
367 {
368 driverPinInNet = i;
369 break;
370 }
371 }
372
373 assert(driverPinInNet >= 0);
374
375 // get the srcPin information
376 auto srcCell = pins[driverPinInNet]->getCell();
377 unsigned int srcCellId = srcCell->getCellId();
378 int driverPathLen = timingNodes[srcCellId]->getLongestPathLength();
379 int driverBackwardLevel = timingNodes[srcCellId]->getBackwardLevel();
380 int driverForwardLevel = timingNodes[srcCellId]->getForwardLevel();
381 auto srcLoc = cellLoc[srcCellId];
382
383 int srcClockRegionXId, srcClockRegionYId;
384 deviceInfo->getClockRegionByLocation(srcLoc.X, srcLoc.Y, srcClockRegionXId, srcClockRegionYId);
385
386 // outfile0 << "handling net:" << curNet->getDesignNet()->getName() << "\n";
387 // outfile0 << "handling driverpin:" << pins[driverPinInNet]->getName()
388 // << " driverBackwardLevel:" << driverBackwardLevel << " locX:" << srcLoc.X << " locY:" << srcLoc.Y
389 // << "\n";
390 if (srcCell->isTimingEndPoint())
391 {
392 // outfile0 << " is TimingEndPoint\n";
393 // calculate the base weight
394 float w = 2 * timingWeight / std::pow((float)(pinNum - 1), 0.5);
395
396 // iterate the sinkPin for evaluation and enhancement
397 for (int pinBeDriven = 0; pinBeDriven < pinNum; pinBeDriven++)
398 {
399 if (pinBeDriven == driverPinInNet)
400 continue;
401
402 // get the sinkPin information
403 auto sinkCell = pins[pinBeDriven]->getCell();
404 unsigned int sinkCellId = sinkCell->getCellId();
405 auto sinkLoc = cellLoc[sinkCellId];
406 int succBackwardLevel = timingNodes[sinkCellId]->getBackwardLevel();
407 // outfile0 << " analyzing pin:" << pins[pinBeDriven]->getName()
408 // << " succBackwardLevel:" << succBackwardLevel << " locX:" << sinkLoc.X << " locY:" <<
409 // sinkLoc.Y
410 // << "\n";
411 // calculate the distance
412 double dis = manhattanDis(srcLoc.X, srcLoc.Y, sinkLoc.X, sinkLoc.Y);
413
414 int sinkClockRegionXId, sinkClockRegionYId;
415 deviceInfo->getClockRegionByLocation(sinkLoc.X, sinkLoc.Y, sinkClockRegionXId, sinkClockRegionYId);
416
417 // dis += std::abs(sinkClockRegionXId - srcClockRegionXId) * 10;
418
419 // handle the potential sinkPins on the longest path by checking the backward level
420 if (succBackwardLevel + 1 >= levelThr)
421 {
422 // the expected distance
423 double disThr = disExpected / (succBackwardLevel + 1);
424 if (dis > disThr)
425 {
426 // enhance the net based on the distance overflow ratio
427 float enhanceRatio = std::pow(dis / disThr, (double)powFactor);
428 // outfile0 << " enhanced by " << enhanceRatio << " dis:" << dis << " disThr:" << disThr
429 // << "\n";
430 curNet->addPseudoNet_enhancePin2Pin(
431 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
432 xSolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, true, false,
433 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
434
435 curNet->addPseudoNet_enhancePin2Pin(
436 ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
437 ySolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, false, true,
438 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
439 }
440 }
441 }
442 }
443 else
444 {
445 // outfile0 << " is NOT TimingEndPoint\n";
446 if (driverPathLen < levelThr)
447 continue;
448
449 // calculate the base weight
450 float w = 2 * timingWeight / std::pow((float)(pinNum - 1), 0.5);
451
452 // iterate the sinkPin for evaluation and enhancement
453 for (int pinBeDriven = 0; pinBeDriven < pinNum; pinBeDriven++)
454 {
455 if (pinBeDriven == driverPinInNet)
456 continue;
457
458 // get the sinkPin information
459 auto sinkCell = pins[pinBeDriven]->getCell();
460 unsigned int sinkCellId = sinkCell->getCellId();
461 auto sinkLoc = cellLoc[sinkCellId];
462 int succPathLen = timingNodes[sinkCellId]->getLongestPathLength();
463 int succBackwardLevel = timingNodes[sinkCellId]->getBackwardLevel();
464 // outfile0 << " analyzing pin:" << pins[pinBeDriven]->getName()
465 // << " succBackwardLevel:" << succBackwardLevel << " locX:" << sinkLoc.X << " locY:" <<
466 // sinkLoc.Y
467 // << "\n";
468 // calculate the distance
469 double dis = manhattanDis(srcLoc.X, srcLoc.Y, sinkLoc.X, sinkLoc.Y);
470
471 int sinkClockRegionXId, sinkClockRegionYId;
472 deviceInfo->getClockRegionByLocation(sinkLoc.X, sinkLoc.Y, sinkClockRegionXId, sinkClockRegionYId);
473
474 // dis += std::abs(sinkClockRegionXId - srcClockRegionXId) * 10;
475
476 // handle the potential sinkPins on the longest path by checking the backward level
477 if (succBackwardLevel >= driverBackwardLevel - 3)
478 {
479 if (timingNodes[sinkCellId]->checkIsRegister() && driverForwardLevel >= driverPathLen * 0.9)
480 {
481 // the expected distance
482 double disThr = disExpected / (driverForwardLevel + 1);
483 if (dis > disThr)
484 {
485 // enhance the net based on the distance overflow ratio
486 float enhanceRatio = std::pow(dis / disThr, powFactor);
487 // outfile0 << " enhanced by " << enhanceRatio << " dis:" << dis << " disThr:" <<
488 // disThr
489 // << "\n";
490 curNet->addPseudoNet_enhancePin2Pin(
491 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
492 xSolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, true, false,
493 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
494
495 curNet->addPseudoNet_enhancePin2Pin(
496 ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
497 ySolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, false, true,
498 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
499 }
500 }
501 else if (succPathLen >= 1 && succPathLen >= driverPathLen * 0.9)
502 {
503 // the expected distance
504 double disThr = disExpected / succPathLen;
505 if (dis > disThr)
506 {
507 // enhance the net based on the distance overflow ratio
508 float enhanceRatio = std::pow(dis / disThr, powFactor);
509 // outfile0 << " enhanced by " << enhanceRatio << " dis:" << dis << " disThr:" <<
510 // disThr
511 // << "\n";
512 curNet->addPseudoNet_enhancePin2Pin(
513 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
514 xSolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, true, false,
515 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
516
517 curNet->addPseudoNet_enhancePin2Pin(
518 ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
519 ySolver->solverData.objectiveVector, w * enhanceRatio, y2xRatio, false, true,
520 PUs[driverPinInNet]->getId(), PUs[pinBeDriven]->getId(), driverPinInNet, pinBeDriven);
521 }
522 }
523 }
524 }
525 }
526 }
527
528 print_status("WirelengthOptimizer: enhanceNetWeight_LevelBased done (maxEnhancedRatio=" +
529 std::to_string(maxEnhanceRatio) + ")");

533void WirelengthOptimizer::addPseudoNet2LoctionForAllPUs(float pesudoNetWeight, bool considerNetNum)
534{
535 int numPUs = placementInfo->getPlacementUnits().size();
536 float minDist = 0.5;
537 float powFactor = placementInfo->getProgress() * 0.45 + 0.5;
538
539 if (considerNetNum)
540 {
541#pragma omp parallel sections
542 {
543#pragma omp section
544 {
545#pragma omp parallel for // schedule(dynamic, 16)
546 for (int PUId = 0; PUId < numPUs; PUId++)
547 {
548 auto curPU = placementInfo->getPlacementUnits()[PUId];
549 bool movable = !curPU->isFixed();
550 if (movable)
551 {
552 float curX = curPU->X();
553 float lastX = curPU->lastX();
554 placementInfo->addPseudoNetsInPlacementInfo(
555 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
556 xSolver->solverData.objectiveVector, curPU, curX,
557 std::pow(curPU->getNetsSetPtr()->size(), powFactor) * pesudoNetWeight /
558 std::max(minDist, std::fabs(lastX - curX)),
559 y2xRatio, true, false);
560 }
561 }
562 }
563#pragma omp section
564 {
565#pragma omp parallel for // schedule(dynamic, 16)
566 for (int PUId = 0; PUId < numPUs; PUId++)
567 {
568 auto curPU = placementInfo->getPlacementUnits()[PUId];
569 bool movable = !curPU->isFixed();
570 if (movable)
571 {
572 float curY = curPU->Y();
573 float lastY = curPU->lastY();
574 placementInfo->addPseudoNetsInPlacementInfo(
575 ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
576 ySolver->solverData.objectiveVector, curPU, curY,
577 std::pow(curPU->getNetsSetPtr()->size(), powFactor) * pesudoNetWeight /
578 std::max(minDist, std::fabs(lastY - curY)),
579 y2xRatio, false, true);
580 }
581 }
582 }

629void WirelengthOptimizer::updatePseudoNetForUserDefinedClusters(float pesudoNetWeight)
630{
631
632 userDefinedClusterFadeOutFactor *= 0.9;
633 std::vector<std::vector<DesignInfo::DesignCell *>> &predefinedCellClusters =
634 placementInfo->getDesignInfo()->getPredefinedClusters();
635
636 std::set<PlacementInfo::PlacementUnit *> reallocatedPUs;
637
638 int reallocateCells = 0;
639 for (auto &clusterCellSet : predefinedCellClusters)
640 {
641 std::set<PlacementInfo::PlacementUnit *> PUsInCurCluster;
642 PUsInCurCluster.clear();
643
644 double avgX = 0;
645 double avgY = 0;
646 int totalCellNum = clusterCellSet.size();
647 if (totalCellNum < 24)
648 continue;
649 float clusterFactor = (200.0 / totalCellNum);
650 if (clusterFactor > 1)
651 clusterFactor = 1;
652 // if (clusterFactor < 0.5)
653 // clusterFactor = 0.333;
654 int totalNetNum = 0;
655
656 float x0 = 10000, y0 = 10000, x1 = -10000, y1 = -10000;
657 for (auto tmpCell : clusterCellSet)
658 {
659 auto tmpPU = placementInfo->getPlacementUnitByCell(tmpCell);
660 if (PUsInCurCluster.find(tmpPU) == PUsInCurCluster.end())
661 {
662 PUsInCurCluster.insert(tmpPU);
663 avgX += tmpPU->X() * tmpPU->getNetsSetPtr()->size();
664 avgY += tmpPU->Y() * tmpPU->getNetsSetPtr()->size();
665 totalNetNum += tmpPU->getNetsSetPtr()->size();
666
667 if (tmpPU->X() < x0)
668 x0 = tmpPU->X();
669 if (tmpPU->X() > x1)
670 x1 = tmpPU->X();
671 if (tmpPU->Y() < y0)
672 y0 = tmpPU->Y();
673 if (tmpPU->Y() > y1)
674 y1 = tmpPU->Y();
675 }
676 }
677
678 if (PUsInCurCluster.size() > 1)
679 {
680 if ((x1 - x0) * (y1 - y0) / PUsInCurCluster.size() < 1 && (x1 - x0) * (y1 - y0) > 1)
681 {
682 avgX /= totalNetNum;
683 avgY /= totalNetNum;
684
685 for (auto tmpPU : PUsInCurCluster)
686 {
687 float fX = avgX;
688 float fY = avgY;
689 placementInfo->legalizeXYInArea(tmpPU, fX, fY);
690
691 placementInfo->addPseudoNetsInPlacementInfo(
692 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
693 xSolver->solverData.objectiveVector, tmpPU, fX,
694 userDefinedClusterFadeOutFactor * clusterFactor * pesudoNetWeight, y2xRatio, true, false);
695
696 placementInfo->addPseudoNetsInPlacementInfo(
697 ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
698 ySolver->solverData.objectiveVector, tmpPU, fY,
699 1.0 / y2xRatio * userDefinedClusterFadeOutFactor * clusterFactor * pesudoNetWeight, y2xRatio,
700 false, true);
701
702 if (reallocatedPUs.find(tmpPU) == reallocatedPUs.end())
703 {
704 reallocatedPUs.insert(tmpPU);
705 reallocateCells += tmpPU->getWeight();
706 }
707 }
708 }
709 }
710 }
711
712 print_warning("set user-defined cluster pseudo net for " + std::to_string(reallocatedPUs.size()) + " PU(s) and " +
713 std::to_string(reallocateCells) + " cell(s).");
714}

716void WirelengthOptimizer::updatePseudoNetForClockRegion(float pesudoNetWeight)
717{
718 if (pesudoNetWeight <= 0)
719 return;
720 auto &PU2ClockRegionCenter = placementInfo->getPU2ClockRegionCenters();
721
722 if (PU2ClockRegionCenter.size() <= 0)
723 return;
724
725 for (auto PUXY : PU2ClockRegionCenter)
726 {
727 auto curPU = PUXY.first;
728 float cX = PUXY.second.first;
729 if (std::fabs(curPU->X() - cX) > 6)
730 placementInfo->addPseudoNetsInPlacementInfo(
731 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
732 xSolver->solverData.objectiveVector, curPU, cX,
733 pesudoNetWeight * std::pow(curPU->getNetsSetPtr()->size(), 1.1), y2xRatio, true, false);
734 else if (std::fabs(curPU->X() - cX) > 3)
735 placementInfo->addPseudoNetsInPlacementInfo(
736 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
737 xSolver->solverData.objectiveVector, curPU, cX, pesudoNetWeight * curPU->getNetsSetPtr()->size(),
738 y2xRatio, true, false);
739 else
740 placementInfo->addPseudoNetsInPlacementInfo(
741 xSolver->solverData.objectiveMatrixTripletList, xSolver->solverData.objectiveMatrixDiag,
742 xSolver->solverData.objectiveVector, curPU, cX,
743 std::fabs(curPU->X() - cX) / 3 * pesudoNetWeight * curPU->getNetsSetPtr()->size(), y2xRatio, true,
744 false);
745 // placementInfo->addPseudoNetsInPlacementInfo(
746 // ySolver->solverData.objectiveMatrixTripletList, ySolver->solverData.objectiveMatrixDiag,
747 // ySolver->solverData.objectiveVector, curPU, cY, pesudoNetWeight * curPU->getNetsSetPtr()->size(),
748 // y2xRatio, false, true);
749 }
750
751 print_warning("update pseudo net of clockt region for " + std::to_string(PU2ClockRegionCenter.size()) + " PUs");
752}
